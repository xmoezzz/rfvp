15#[derive(Clone, Debug)]
16pub enum Stmt {
17    Simple(String),
18    Return(Option<String>),
19    Break,
20    If {
21        arms: Vec<(String, Vec<Stmt>)>,
22        else_arm: Option<Vec<Stmt>>,
23    },
24    While {
25        cond: String,
26        body: Vec<Stmt>,
27    },

30fn is_comment_or_empty(s: &str) -> bool {
31    let t = s.trim();
32    t.is_empty() || t.starts_with("--")
33}

35fn strip_local(stmt: &str) -> &str {
36    let t = stmt.trim_start();
37    if let Some(rest) = t.strip_prefix("local ") {
38        return rest.trim_start();
39    }
40    t
41}

43fn is_if_start(s: &str) -> bool {
44    let t = s.trim();
45    t.starts_with("if ") && t.ends_with(" then")
46}
47
48fn is_elseif(s: &str) -> bool {
49    let t = s.trim();
50    t.starts_with("elseif ") && t.ends_with(" then")
51}
52
53fn is_else(s: &str) -> bool {
54    s.trim() == "else"
55}
56
57fn is_while_start(s: &str) -> bool {
58    let t = s.trim();
59    t.starts_with("while ") && t.ends_with(" do")
60}
61
62fn is_end(s: &str) -> bool {
63    s.trim() == "end"
64}

66fn extract_if_cond(line: &str) -> Result<String> {
67    let t = line.trim();
68    if !t.ends_with(" then") {
69        bail!("invalid if header: {t}");
70    }
71    let inner = t
72        .strip_prefix("if ")
73        .or_else(|| t.strip_prefix("elseif "))
74        .ok_or_else(|| anyhow!("invalid if header: {t}"))?;
75    let inner = inner.trim_end_matches(" then");
76    Ok(inner.trim().to_string())
77}
78
79fn extract_while_cond(line: &str) -> Result<String> {
80    let t = line.trim();
81    if !t.ends_with(" do") {
82        bail!("invalid while header: {t}");
83    }
84    let inner = t
85        .strip_prefix("while ")
86        .ok_or_else(|| anyhow!("invalid while header: {t}"))?;
87    let inner = inner.trim_end_matches(" do");
88    Ok(inner.trim().to_string())
89}

91fn parse_block(lines: &[String], i: &mut usize, stop_on: &[&str]) -> Result<Vec<Stmt>> {
92    let mut out: Vec<Stmt> = Vec::new();
93
94    while *i < lines.len() {
95        let line0 = lines[*i].clone();
96        let line = line0.trim();
97
98        if is_comment_or_empty(line) {
99            *i += 1;
100            continue;
101        }
102
103        if stop_on.iter().any(|tok| match *tok {
104            "elseif" => is_elseif(line),
105            "else" => is_else(line),
106            "end" => is_end(line),
107            _ => false,
108        }) {
109            break;
110        }
111
112        if is_if_start(line) {
113            let cond = extract_if_cond(line)?;
114            *i += 1;
115            let then_block = parse_block(lines, i, &["elseif", "else", "end"])?;
116
117            let mut arms: Vec<(String, Vec<Stmt>)> = vec![(cond, then_block)];
118            while *i < lines.len() && is_elseif(lines[*i].trim()) {
119                let c = extract_if_cond(lines[*i].trim())?;
120                *i += 1;
121                let b = parse_block(lines, i, &["elseif", "else", "end"])?;
122                arms.push((c, b));
123            }
124
125            let mut else_arm: Option<Vec<Stmt>> = None;
126            if *i < lines.len() && is_else(lines[*i].trim()) {
127                *i += 1;
128                let b = parse_block(lines, i, &["end"])?;
129                else_arm = Some(b);
130            }
131
132            if *i >= lines.len() || !is_end(lines[*i].trim()) {
133                bail!("if without closing end");
134            }
135            *i += 1;
136
137            out.push(Stmt::If { arms, else_arm });
138            continue;
139        }
140
141        if is_while_start(line) {
142            let cond = extract_while_cond(line)?;
143            *i += 1;
144            let body = parse_block(lines, i, &["end"])?;
145            if *i >= lines.len() || !is_end(lines[*i].trim()) {
146                bail!("while without closing end");
147            }
148            *i += 1;
149            out.push(Stmt::While { cond, body });
150            continue;
151        }
152
153        if line == "break" {
154            *i += 1;
155            out.push(Stmt::Break);
156            continue;
157        }
158
159        if line == "return" {
160            *i += 1;
161            out.push(Stmt::Return(None));
162            continue;
163        }
164
165        if let Some(rest) = line.strip_prefix("return ") {
166            *i += 1;
167            out.push(Stmt::Return(Some(rest.trim().to_string())));
168            continue;
169        }
170
171        // Local declarations emitted by the decompiler are not semantic (locals are inferred).
172        // Examples:
173        //   local S0, S1, S2
174        //   local __ret = nil
175        // We skip pure declarations and keep assignments.
176        if line.starts_with("local ") && !line.contains('=') {
177            *i += 1;
178            continue;
179        }
180
181        // Treat everything else as a simple IR line.
182        let simple = strip_local(line).to_string();
183        *i += 1;
184        out.push(Stmt::Simple(simple));
185    }
186
187    Ok(out)
188}

190fn split_functions(lua_text: &str) -> Result<Vec<Vec<String>>> {
191    let lines: Vec<String> = lua_text.lines().map(|s| s.to_string()).collect();
192    let head_re = Regex::new(r"^(?:local\s+)?function\s+").unwrap();
193
194    let mut out: Vec<Vec<String>> = Vec::new();
195    let mut i = 0usize;
196    while i < lines.len() {
197        if head_re.is_match(lines[i].trim()) {
198            let start = i;
199            let mut nest = 1i32;
200            i += 1;
201            while i < lines.len() && nest > 0 {
202                let t = lines[i].trim();
203                if head_re.is_match(t) {
204                    nest += 1;
205                } else if is_if_start(t) {
206                    // IMPORTANT: `elseif` does not introduce a new block that needs its own `end`.
207                    // It is part of the same `if ... then ... elseif ... then ... end` construct.
208                    // Counting it as a new nested block breaks function splitting and can cause
209                    // later function definitions to be swallowed into the current function.
210                    nest += 1;
211                } else if is_while_start(t) {
212                    nest += 1;
213                } else if is_end(t) {
214                    nest -= 1;
215                }
216                i += 1;
217            }
218            out.push(lines[start..i].to_vec());
219        } else {
220            i += 1;
221        }
222    }
223
224    if out.is_empty() {
225        bail!("no functions found in Lua");
226    }
227
228    Ok(out)
229}

231pub fn parse_lua(path: &Path) -> Result<Vec<Function>> {
232    let txt = std::fs::read_to_string(path).with_context(|| format!("read lua: {}", path.display()))?;
233    let funcs_lines = split_functions(&txt)?;
234
235    let head_re = Regex::new(r"^(?:local\s+)?function\s+([A-Za-z_]\w*)\s*\(([^)]*)\)\s*$").unwrap();
236    let re_a = Regex::new(r"\ba(\d+)\b").unwrap();
237    let re_l = Regex::new(r"\bl(\d+)\b").unwrap();
238
239    let mut funs: Vec<Function> = Vec::new();
240    for fl in funcs_lines {
241        if fl.is_empty() {
242            continue;
243        }
244        let head = fl[0].trim();
245        let caps = head_re
246            .captures(head)
247            .ok_or_else(|| anyhow!("unexpected function header: {head}"))?;
248        let name = caps.get(1).unwrap().as_str().to_string();
249        let args_s = caps.get(2).unwrap().as_str().trim();
250        let args: Vec<&str> = if args_s.is_empty() {
251            vec![]
252        } else {
253            args_s.split(',').map(|x| x.trim()).filter(|x| !x.is_empty()).collect()
254        };
255
256        // args_count: max aN seen in header or body + 1.
257        let mut max_a: Option<u32> = None;
258        for a in &args {
259            if let Some(mm) = re_a.captures(a) {
260                let v: u32 = mm.get(1).unwrap().as_str().parse().unwrap_or(0);
261                max_a = Some(max_a.map(|x| x.max(v)).unwrap_or(v));
262            }
263        }
264        for ln in &fl {
265            for mm in re_a.captures_iter(ln) {
266                let v: u32 = mm.get(1).unwrap().as_str().parse().unwrap_or(0);
267                max_a = Some(max_a.map(|x| x.max(v)).unwrap_or(v));
268            }
269        }
270        let args_count = i8::try_from(max_a.map(|x| x + 1).unwrap_or(0))
271            .map_err(|_| anyhow!("args_count does not fit i8"))?;
272
273        // locals_count: max lN seen in body + 1.
274        let mut max_l: Option<u32> = None;
275        for ln in &fl {
276            for mm in re_l.captures_iter(ln) {
277                let v: u32 = mm.get(1).unwrap().as_str().parse().unwrap_or(0);
278                max_l = Some(max_l.map(|x| x.max(v)).unwrap_or(v));
279            }
280        }
281        let locals_count = i8::try_from(max_l.map(|x| x + 1).unwrap_or(0))
282            .map_err(|_| anyhow!("locals_count does not fit i8"))?;
283
284        // Body lines: between header and the final end.
285        if fl.len() < 2 {
286            bail!("function {name}: too short");
287        }
288        let body_lines: Vec<String> = fl[1..fl.len() - 1].to_vec();
289        let mut idx = 0usize;
290        let body = parse_block(&body_lines, &mut idx, &[])?;
291
292        funs.push(Function {
293            name,
294            args_count,
295            locals_count,
296            body,
297            raw: fl,
298        });
299    }
300
301    Ok(funs)
