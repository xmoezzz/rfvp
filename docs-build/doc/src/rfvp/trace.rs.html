5/// Trace categories, enabled via environment variables.
6///
7/// Supported:
8/// - RFVP_TRACE="vm,syscall,prim,prim_tree,motion,render" (comma/space separated; "all" enables all)
9/// - RFVP_TRACE_VM=1, RFVP_TRACE_SYSCALL=1, RFVP_TRACE_PRIM=1, RFVP_TRACE_PRIM_TREE=1,
10///   RFVP_TRACE_MOTION=1, RFVP_TRACE_RENDER=1
11///
12/// Rate limits / caps:
13/// - RFVP_TRACE_PRIM_TREE_EVERY (u64, default 60)
14/// - RFVP_TRACE_MOTION_EVERY (u64, default 30)
15/// - RFVP_TRACE_PRIM_TREE_MAX_NODES (usize, default 256)
16/// - RFVP_TRACE_PRIM_TREE_MAX_DEPTH (usize, default 32)
17/// - RFVP_TRACE_MOTION_MAX (usize, default 16)
18#[derive(Clone, Copy, Debug, PartialEq, Eq)]
19pub enum TraceKind {
20    Vm,
21    Syscall,
22    Prim,
23    PrimTree,
24    Motion,
25    Render,
26}
27
28#[derive(Clone, Copy, Debug)]
29struct TraceConfig {
30    mask: u32,
31    prim_tree_every: u64,
32    motion_every: u64,
33    prim_tree_max_nodes: usize,
34    prim_tree_max_depth: usize,
35    motion_max: usize,
36}
37
38const M_VM: u32 = 1 << 0;
39const M_SYSCALL: u32 = 1 << 1;
40const M_PRIM: u32 = 1 << 2;
41const M_PRIM_TREE: u32 = 1 << 3;
42const M_MOTION: u32 = 1 << 4;
43const M_RENDER: u32 = 1 << 5;
44
45fn parse_bool_env(name: &str) -> bool {
46    match env::var(name) {
47        Ok(v) => {
48            let s = v.trim().to_ascii_lowercase();
49            !(s.is_empty() || s == "0" || s == "false" || s == "no" || s == "off")
50        }
51        Err(_) => false,
52    }
53}
54
55fn parse_u64_env(name: &str, default_v: u64) -> u64 {
56    env::var(name).ok().and_then(|v| v.trim().parse::<u64>().ok()).unwrap_or(default_v)
57}
58
59fn parse_usize_env(name: &str, default_v: usize) -> usize {
60    env::var(name).ok().and_then(|v| v.trim().parse::<usize>().ok()).unwrap_or(default_v)
61}

63fn parse_mask_from_trace_list(s: &str) -> u32 {
64    let mut mask = 0u32;
65    for raw in s.split(|c: char| c == ',' || c == ';' || c.is_whitespace()) {
66        let t = raw.trim().to_ascii_lowercase();
67        if t.is_empty() {
68            continue;
69        }
70        match t.as_str() {
71            "all" => {
72                mask |= M_VM | M_SYSCALL | M_PRIM | M_PRIM_TREE | M_MOTION | M_RENDER;
73            }
74            "vm" => mask |= M_VM,
75            "syscall" | "sc" => mask |= M_SYSCALL,
76            "prim" => mask |= M_PRIM,
77            "prim_tree" | "tree" => mask |= M_PRIM_TREE,
78            "motion" => mask |= M_MOTION,
79            "render" => mask |= M_RENDER,
80            _ => {}
81        }
82    }
83    mask
84}

86fn build_config() -> TraceConfig {
87    let mut mask = 0u32;
88
89    if let Ok(list) = env::var("RFVP_TRACE") {
90        mask |= parse_mask_from_trace_list(&list);
91    }
92
93    if parse_bool_env("RFVP_TRACE_VM") {
94        mask |= M_VM;
95    }
96    if parse_bool_env("RFVP_TRACE_SYSCALL") {
97        mask |= M_SYSCALL;
98    }
99    if parse_bool_env("RFVP_TRACE_PRIM") {
100        mask |= M_PRIM;
101    }
102    if parse_bool_env("RFVP_TRACE_PRIM_TREE") {
103        mask |= M_PRIM_TREE;
104    }
105    if parse_bool_env("RFVP_TRACE_MOTION") {
106        mask |= M_MOTION;
107    }
108    if parse_bool_env("RFVP_TRACE_RENDER") {
109        mask |= M_RENDER;
110    }
111
112    TraceConfig {
113        mask,
114        prim_tree_every: parse_u64_env("RFVP_TRACE_PRIM_TREE_EVERY", 60),
115        motion_every: parse_u64_env("RFVP_TRACE_MOTION_EVERY", 30),
116        prim_tree_max_nodes: parse_usize_env("RFVP_TRACE_PRIM_TREE_MAX_NODES", 256),
117        prim_tree_max_depth: parse_usize_env("RFVP_TRACE_PRIM_TREE_MAX_DEPTH", 32),
118        motion_max: parse_usize_env("RFVP_TRACE_MOTION_MAX", 16),
119    }
120}
121
122fn cfg() -> &'static TraceConfig {
123    static CFG: OnceLock<TraceConfig> = OnceLock::new();
124    CFG.get_or_init(build_config)
125}
126
127pub fn enabled(k: TraceKind) -> bool {
128    let c = cfg();
129    match k {
130        TraceKind::Vm => (c.mask & M_VM) != 0,
131        TraceKind::Syscall => (c.mask & M_SYSCALL) != 0,
132        TraceKind::Prim => (c.mask & M_PRIM) != 0,
133        TraceKind::PrimTree => (c.mask & M_PRIM_TREE) != 0,
134        TraceKind::Motion => (c.mask & M_MOTION) != 0,
135        TraceKind::Render => (c.mask & M_RENDER) != 0,
136    }
137}

155pub fn should_dump_prim_tree(frame_no: u64) -> bool {
156    enabled(TraceKind::PrimTree) && (prim_tree_every() == 0 || frame_no % prim_tree_every() == 0)
157}

159pub fn should_dump_motion(frame_no: u64) -> bool {
160    enabled(TraceKind::Motion) && (motion_every() == 0 || frame_no % motion_every() == 0)
161}
