5fn should_skip_update(prim_manager: &PrimManager, prim_id: u32, flag: bool) -> bool {
6    let custom_root_id = prim_manager.get_custom_root_prim_id();
7
8    if flag {
9        // Custom-root traversal gate.
10        if custom_root_id == 0 {
11            return true;
12        }
13        // Require the primitive to have a parent chain reaching the custom root.
14        let mut cur: i16 = prim_id as i16;
15        let parent0 = prim_manager.get_prim_immutable(cur).get_parent();
16        if parent0 == INVAILD_PRIM_HANDLE {
17            return true;
18        }
19        let mut next = parent0;
20        while next != INVAILD_PRIM_HANDLE {
21            if next as u16 == custom_root_id {
22                return false;
23            }
24            next = prim_manager.get_prim_immutable(next).get_parent();
25        }
26        return true;
27    }
28
29    // Pause gate: if self or any ancestor is paused, skip updating but keep motion running.
30    let mut cur: i16 = prim_id as i16;
31    loop {
32        let p = prim_manager.get_prim_immutable(cur);
33        if p.get_paused() {
34            return true;
35        }
36        let parent = p.get_parent();
37        drop(p);
38        if parent == INVAILD_PRIM_HANDLE {
39            break;
40        }
41        cur = parent;
42    }
43
44    false
45}

48pub enum ScaleMotionType {
49    None = 0,
50    Linear,
51    Accelerate,
52    Decelerate,
53    Rebound,
54    Bounce,
55}
56
57impl TryFrom<i32> for ScaleMotionType {
58    type Error = ();
59
60    fn try_from(value: i32) -> Result<Self, Self::Error> {
61        match value {
62            0 => Ok(ScaleMotionType::None),
63            1 => Ok(ScaleMotionType::Linear),
64            2 => Ok(ScaleMotionType::Accelerate),
65            3 => Ok(ScaleMotionType::Decelerate),
66            4 => Ok(ScaleMotionType::Rebound),
67            5 => Ok(ScaleMotionType::Bounce),
68            _ => Err(()),
69        }
70    }
71}

74pub struct ScaleMotion {
75    id: u16,
76    prim_id: u32,
77    running: bool,
78    reverse: bool,
79    src_fx: i16,
80    src_fy: i16,
81    dst_fx: i16,
82    dst_fy: i16,
83    duration: i32,
84    elapsed: i32,
85    typ: ScaleMotionType,
86}
87
88impl ScaleMotion {
89    pub fn new(id: u16) -> ScaleMotion {
90        ScaleMotion {
91            id,
92            prim_id: 0,
93            running: false,
94            reverse: false,
95            src_fx: 1000,
96            src_fy: 1000,
97            dst_fx: 1000,
98            dst_fy: 1000,
99            duration: 0,
100            elapsed: 0,
101            typ: ScaleMotionType::None,
102        }
103    }
104
105    fn finish(&mut self) {
106        self.running = false;
107        self.typ = ScaleMotionType::None;
108        self.prim_id = 0;
109        self.elapsed = 0;
110        self.duration = 0;
111    }
112
113    pub fn is_running_for(&self, prim_id: u32) -> bool {
114        self.running && self.prim_id == prim_id && self.typ != ScaleMotionType::None
115    }
116
117    pub fn update(&mut self, prim_manager: &PrimManager, flag: bool, elapsed_delta: i32) -> bool {
118        if self.typ == ScaleMotionType::None || self.prim_id as i16 == INVAILD_PRIM_HANDLE {
119            return true;
120        }
121
122        if should_skip_update(prim_manager, self.prim_id, flag) {
123            return true;
124        }
125
126        let mut dt = elapsed_delta;
127        if self.reverse && dt < 0 {
128            dt = -dt;
129        }
130        self.elapsed += dt;
131
132        if dt < 0 || self.elapsed >= self.duration {
133            let mut prim = prim_manager.get_prim(self.prim_id as i16);
134            prim.apply_attr(0x40);
135            prim.set_factor_x(self.dst_fx);
136            prim.set_factor_y(self.dst_fy);
137            return false;
138        }
139
140        let src_x = self.src_fx as i64;
141        let src_y = self.src_fy as i64;
142        let dst_x = self.dst_fx as i64;
143        let dst_y = self.dst_fy as i64;
144        let delta_x = dst_x - src_x;
145        let delta_y = dst_y - src_y;
146        let e = self.elapsed as i64;
147        let d = self.duration as i64;
148
149        let (fx, fy) = match self.typ {
150            ScaleMotionType::Linear => (src_x + delta_x * e / d, src_y + delta_y * e / d),
151            ScaleMotionType::Accelerate => (
152                src_x + delta_x * e * e / (d * d),
153                src_y + delta_y * e * e / (d * d),
154            ),
155            ScaleMotionType::Decelerate => (
156                dst_x - delta_x * (d - e) * (d - e) / (d * d),
157                dst_y - delta_y * (d - e) * (d - e) / (d * d),
158            ),
159            ScaleMotionType::Rebound => {
160                let half_delta_x = delta_x / 2;
161                let half_delta_y = delta_y / 2;
162                let half_dur = d / 2;
163                if e > half_dur {
164                    let denom = (d - half_dur) * (d - half_dur);
165                    let t = (d - e) * (d - e);
166                    (
167                        dst_x - (delta_x - half_delta_x) * t / denom,
168                        dst_y - (delta_y - half_delta_y) * t / denom,
169                    )
170                } else {
171                    let denom = half_dur * half_dur;
172                    (
173                        src_x + half_delta_x * e * e / denom,
174                        src_y + half_delta_y * e * e / denom,
175                    )
176                }
177            }
178            ScaleMotionType::Bounce => {
179                let half_delta_x = delta_x / 2;
180                let half_delta_y = delta_y / 2;
181                let half_dur = d / 2;
182                if e > half_dur {
183                    let t = (e - half_dur) * (e - half_dur);
184                    let denom = (d - half_dur) * (d - half_dur);
185                    (
186                        half_delta_x + src_x + (delta_x - half_delta_x) * t / denom,
187                        half_delta_y + src_y + (delta_y - half_delta_y) * t / denom,
188                    )
189                } else {
190                    let t = (half_dur - e) * (half_dur - e);
191                    let denom = half_dur * half_dur;
192                    (
193                        half_delta_x + src_x - half_delta_x * t / denom,
194                        half_delta_y + src_y - half_delta_y * t / denom,
195                    )
196                }
197            }
198            _ => (dst_x, dst_y),
199        };
200
201        let mut prim = prim_manager.get_prim(self.prim_id as i16);
202        prim.apply_attr(0x40);
203        prim.set_factor_x(fx as i16);
204        prim.set_factor_y(fy as i16);
205        true
206    }
207}

209pub struct ScaleMotionContainer {
210    motions: Vec<ScaleMotion>,
211    free_ids: Vec<u16>,
212}
213
214impl ScaleMotionContainer {
215    pub fn new() -> ScaleMotionContainer {
216        let mut motions = Vec::with_capacity(512);
217        for i in 0..512u16 {
218            motions.push(ScaleMotion::new(i));
219        }
220        let free_ids: Vec<u16> = (0..512u16).rev().collect();
221        ScaleMotionContainer { motions, free_ids }
222    }
223
224    pub fn push_motion(
225        &mut self,
226        prim_id: u32,
227        src_w_factor: i32,
228        src_h_factor: i32,
229        dst_w_factor: i32,
230        dst_h_factor: i32,
231        duration: i32,
232        typ: ScaleMotionType,
233        reverse: bool,
234    ) -> Result<()> {
235        let _ = self.stop_motion(prim_id);
236        let Some(id) = self.free_ids.pop() else { return Ok(()); };
237        let m = &mut self.motions[id as usize];
238        m.prim_id = prim_id;
239        m.running = true;
240        m.reverse = reverse;
241        m.src_fx = src_w_factor as i16;
242        m.src_fy = src_h_factor as i16;
243        m.dst_fx = dst_w_factor as i16;
244        m.dst_fy = dst_h_factor as i16;
245        m.duration = duration;
246        m.elapsed = 0;
247        m.typ = typ;
248        Ok(())
249    }
250
251    pub fn stop_motion(&mut self, prim_id: u32) -> Result<()> {
252        for m in &mut self.motions {
253            if m.is_running_for(prim_id) {
254                let id = m.id;
255                m.finish();
256                self.free_ids.push(id);
257                break;
258            }
259        }
260        Ok(())
261    }
262
263    pub fn test_motion(&self, prim_id: u32) -> bool {
264        self.motions.iter().any(|m| m.is_running_for(prim_id))
265    }
266
267    pub fn exec_scale_motion(&mut self, prim_manager: &PrimManager, flag: bool, elapsed: i32) {
268        for i in 0..self.motions.len() {
269            if !self.motions[i].running {
270                continue;
271            }
272            if !self.motions[i].update(prim_manager, flag, elapsed) {
273                let id = self.motions[i].id;
274                self.motions[i].finish();
275                self.free_ids.push(id);
276            }
277        }
278    }
279
280    pub fn debug_dump(&self, max: usize) -> String {
281        let mut out = String::new();
282        let mut n = 0usize;
283        for m in &self.motions {
284            if !m.running {
285                continue;
286            }
287            if n >= max {
288                break;
289            }
290            out.push_str(&format!(
291                "  [rot] prim={} src={}:{} dst={}:{} elapsed={} dur={} type={:?} rev={}
292",
293                m.prim_id, m.src_fx, m.src_fy, m.dst_fx, m.dst_fy, m.elapsed, m.duration, m.typ, m.reverse
294            ));
295            n += 1;
296        }
297        out
298    }
299
300    pub fn debug_running_count(&self) -> usize {
301        self.motions.iter().filter(|m| m.running).count()
302    }
