1/// Second dissolve system (engine-internal overlay fade).
2///
3/// This is intentionally separate from the script-controlled dissolve (mask/colored dissolve1).
4/// The original engine uses a small mode machine:
5///   0: off
6///   1: hold (opaque)
7///   2: fade-out (opaque -> transparent)
8///   3: fade-in  (transparent -> opaque)
9///
10/// This module provides a small state machine with time-based alpha update.
11#[derive(Debug, Clone)]
12pub(crate) struct Dissolve2State {
13    mode: u8,
14    color_id: u32,
15    duration_ms: u32,
16    elapsed_ms: u32,
17    alpha: f32,
18
19    /// When true, after fade-in reaches opaque, automatically switch to fade-out
20    /// on the next tick (used for load/save "black then reveal").
21    pending_fade_out: bool,
22}
23
24#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
25pub(crate) struct Dissolve2SnapshotV1 {
26    pub mode: u8,
27    pub color_id: u32,
28    pub duration_ms: u32,
29    pub elapsed_ms: u32,
30    pub alpha: f32,
31    pub pending_fade_out: bool,
32}
33
34impl Dissolve2State {
35    pub(crate) fn new() -> Self {
36        Self {
37            mode: 0,
38            color_id: 0,
39            duration_ms: 1,
40            elapsed_ms: 0,
41            alpha: 0.0,
42            pending_fade_out: false,
43        }
44    }
45
46    pub(crate) fn mode(&self) -> u8 {
47        self.mode
48    }
49
50    pub(crate) fn color_id(&self) -> u32 {
51        self.color_id
52    }
53
54    pub(crate) fn alpha(&self) -> f32 {
55        self.alpha
56    }
57
58    pub(crate) fn is_transitioning(&self) -> bool {
59        self.mode == 2 || self.mode == 3
60    }
61
62    pub(crate) fn is_active(&self) -> bool {
63        self.mode != 0
64    }
65
66    pub(crate) fn reset(&mut self) {
67        self.mode = 0;
68        self.color_id = 0;
69        self.duration_ms = 1;
70        self.elapsed_ms = 0;
71        self.alpha = 0.0;
72        self.pending_fade_out = false;
73    }
74
75    pub(crate) fn start_hold(&mut self, color_id: u32) {
76        self.mode = 1;
77        self.color_id = color_id;
78        self.duration_ms = 1;
79        self.elapsed_ms = 0;
80        self.alpha = 1.0;
81        self.pending_fade_out = false;
82    }
83
84    pub(crate) fn start_fade_in(&mut self, color_id: u32, duration_ms: u32) {
85        self.mode = 3;
86        self.color_id = color_id;
87        self.duration_ms = duration_ms.max(1);
88        self.elapsed_ms = 0;
89        self.alpha = 0.0;
90        self.pending_fade_out = false;
91    }
92
93    pub(crate) fn start_fade_out(&mut self, duration_ms: u32) {
94        // Fade-out keeps previous color_id (expected behavior for load/save).
95        self.mode = 2;
96        self.duration_ms = duration_ms.max(1);
97        self.elapsed_ms = 0;
98        self.alpha = 1.0;
99        self.pending_fade_out = false;
100    }
101
102    /// Convenience for "fade to opaque then reveal", matching the original load-style usage.
103    pub(crate) fn start_in_out(&mut self, color_id: u32, duration_ms: u32) {
104        self.start_fade_in(color_id, duration_ms);
105        self.pending_fade_out = true;
106    }
107
108    /// Advance the state by elapsed time.
109    ///
110    /// If `elapsed_ms` is `u32::MAX`, we treat it as fast-forward and finish the transition
111    /// immediately (used by Ctrl/ControlPulse semantics in the engine).
112    pub(crate) fn tick(&mut self, elapsed_ms: u32) {
113        if self.mode == 0 {
114            self.alpha = 0.0;
115            self.pending_fade_out = false;
116            return;
117        }
118
119        // Hold: stay opaque until externally changed.
120        if self.mode == 1 {
121            self.alpha = 1.0;
122            self.pending_fade_out = false;
123            return;
124        }
125
126        // Transitioning: 2 (fade-out) or 3 (fade-in)
127        if elapsed_ms == u32::MAX {
128            self.elapsed_ms = self.duration_ms;
129        } else {
130            self.elapsed_ms = self.elapsed_ms.saturating_add(elapsed_ms);
131        }
132
133        let dur = self.duration_ms.max(1);
134        let t = (self.elapsed_ms as f32 / dur as f32).clamp(0.0, 1.0);
135
136        match self.mode {
137            3 => {
138                // fade-in: transparent -> opaque
139                self.alpha = t;
140                if self.elapsed_ms >= dur {
141                    self.alpha = 1.0;
142                    if self.pending_fade_out {
143                        // Switch to fade-out after reaching opaque.
144                        self.pending_fade_out = false;
145                        self.mode = 2;
146                        self.elapsed_ms = 0;
147                        // keep alpha at 1.0 for the first frame of fade-out
148                        self.alpha = 1.0;
149                    } else {
150                        // become hold
151                        self.mode = 1;
152                        self.alpha = 1.0;
153                    }
154                }
155            }
156            2 => {
157                // fade-out: opaque -> transparent
158                self.alpha = 1.0 - t;
159                if self.elapsed_ms >= dur {
160                    self.mode = 0;
161                    self.alpha = 0.0;
162                    self.pending_fade_out = false;
163                }
164            }
165            _ => {
166                // Defensive: unknown mode -> off
167                self.mode = 0;
168                self.alpha = 0.0;
169                self.pending_fade_out = false;
170            }
171        }
172    }

174    pub(crate) fn capture_snapshot_v1(&self) -> Dissolve2SnapshotV1 {

178            duration_ms: self.duration_ms,

182        }
183    }
184
185    pub(crate) fn apply_snapshot_v1(&mut self, snap: &Dissolve2SnapshotV1) {
186        self.mode = snap.mode;
187        self.color_id = snap.color_id;
188        self.duration_ms = snap.duration_ms.max(1);
189        self.elapsed_ms = snap.elapsed_ms;
190        self.alpha = snap.alpha;
191        self.pending_fade_out = snap.pending_fade_out;
192    }
193}
