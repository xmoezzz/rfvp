1/// Equivalent to `ff_emulated_edge_mc()` for 8-bit.
2///
3/// Copies a rectangle from `src` into `buf`, replicating edges when `src_x/src_y`
4/// are outside the image bounds.
5pub fn emulated_edge_mc(
6    buf: &mut [u8],
7    src: &[u8],
8    buf_linesize: isize,
9    src_linesize: isize,
10    block_w: usize,
11    block_h: usize,
12    mut src_x: isize,
13    mut src_y: isize,
14    w: isize,
15    h: isize,
16) {
17    if w == 0 || h == 0 {
18        return;
19    }
20
21    // Clamp starting point similar to upstream.
22    if src_y >= h {
23        src_y = h - 1;
24    } else if src_y <= -(block_h as isize) {
25        src_y = 1 - block_h as isize;
26    }
27    if src_x >= w {
28        src_x = w - 1;
29    } else if src_x <= -(block_w as isize) {
30        src_x = 1 - block_w as isize;
31    }
32
33    let start_y = 0.max(-src_y) as usize;
34    let start_x = 0.max(-src_x) as usize;
35    let end_y = (block_h as isize).min(h - src_y) as usize;
36    let end_x = (block_w as isize).min(w - src_x) as usize;
37
38    let copy_w = end_x - start_x;
39
40    // Helper to read a pixel from src with absolute coordinates.
41    let src_at = |x: isize, y: isize| -> u8 {
42        let x = x.clamp(0, w - 1) as usize;
43        let y = y.clamp(0, h - 1) as usize;
44        src[y * (src_linesize as usize) + x]
45    };
46
47    // Fill rows.
48    for by in 0..block_h {
49        let yy = src_y + by as isize;
50        for bx in 0..block_w {
51            let xx = src_x + bx as isize;
52            let v = src_at(xx, yy);
53            let dst_index = by * (buf_linesize.unsigned_abs() as usize) + bx;
54            buf[dst_index] = v;
55        }
56    }
57
58    // The upstream implementation first copies valid interior region then extends
59    // left/right. The above direct clamp per-pixel yields identical results.
60    let _ = (start_y, start_x, end_y, end_x, copy_w); // keep variables referenced
61}
