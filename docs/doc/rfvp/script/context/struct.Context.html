pub struct Context { /* private fields */ }

implementation of the virtual machine stack layout: |—————–| | arg(n) | <- … | … | | arg(0) | <- -2 (0xfe in hex) | SavedFrameInfo | <- -1, includes the base/current pointer and the return address |—————–| | local(0) | <- cur_stack_base
