10/// Main configuration for the game window
11/// Please use [`WindowConfigBuilder`] if you want to build if from code.
12#[derive(Debug, Serialize, Deserialize, Clone)]
13pub struct WindowConfig {
14    /// Enables fullscreen mode
15    pub(crate) fullscreen: bool,
16    /// Default window width and height in pixels.
17    pub(crate) dimensions: Option<(u32, u32)>,
18    /// Minimum window width and height in pixels.
19    pub(crate) min_dimensions: Option<(u32, u32)>,
20    /// Maximum window width and height in pixels.
21    pub(crate) max_dimensions: Option<(u32, u32)>,
22    /// Whether to display the window, Use full for loading
23    pub(crate) visibility: bool,
24    /// The path relative to the game executable of the window icon.
25    pub(crate) icon: Option<PathBuf>,
26    /// Whether the window should always be on top of other windows.
27    pub(crate) always_on_top: bool,
28    /// Whether the window should have borders and bars.
29    pub(crate) decorations: bool,
30    /// Whether the window should be maximized upon creation.
31    pub(crate) maximized: bool,
32    /// If the user can resize the window
33    pub(crate) resizable: bool,
34    /// If the window should be able to be transparent.
35    pub(crate) transparent: bool,
36}
37
38impl Default for WindowConfig {
39    fn default() -> Self {
40        Self {
41            fullscreen: false,
42            dimensions: Some((1024, 768)),
43            min_dimensions: Some((384, 336)),
44            max_dimensions: None,
45            visibility: true,
46            icon: None,
47            always_on_top: false,
48            decorations: true,
49            maximized: false,
50            resizable: true,
51            transparent: false,
52        }
53    }
54}

56impl WindowConfig {
57    pub(crate) fn into(self, config: &AppConfig) -> WindowAttributes {
58        let mut builder = WindowAttributes::default();
59
60        builder = builder.with_title(config.app_name.clone())
61            .with_fullscreen(None);
62        if self.dimensions.is_some() {
63            builder = builder.with_inner_size(self.dimensions.map(|d| LogicalSize::new(d.0, d.1)).unwrap())
64        }
65        if self.min_dimensions.is_some() {
66            builder = builder.with_min_inner_size(self.min_dimensions.map(|d| LogicalSize::new(d.0, d.1)).unwrap())
67        }
68        if self.max_dimensions.is_some() {
69            builder = builder.with_max_inner_size(self.max_dimensions.map(|d| LogicalSize::new(d.0, d.1)).unwrap())
70        }
71
72        builder
73            .with_visible(self.visibility)
74            .with_window_icon(None)
75            .with_window_level(if self.always_on_top { WindowLevel::AlwaysOnTop } else { WindowLevel::Normal })
76            .with_decorations(self.decorations)
77            .with_maximized(self.maximized)
78            .with_resizable(self.resizable)
79            .with_transparent(self.transparent)
80    }
81}

83/// `WindowConfigBuilder` is a convenience builder to create a `WindowConfig` from code.
84pub struct WindowConfigBuilder {
85    config: WindowConfig,
86}
87
88impl WindowConfigBuilder {
89    /// Create a new `WindowConfigBuilder` builder
90    pub fn new() -> Self {
91        Self { config: Default::default() }
92    }
93
94    /// Dimension of the window
95    pub fn with_dimensions(mut self, dimensions: (u32, u32)) -> Self {
96        self.config.dimensions = Some(dimensions);
97        self
98    }
99
100    /// Whether or not the window should be resizable
101    pub fn with_resizable(mut self, resizable: bool) -> Self {
102        self.config.resizable = resizable;
103        self
104    }
105
106    /// Retrieves the configuration built
107    pub fn get(self) -> WindowConfig {
108        self.config
109    }
