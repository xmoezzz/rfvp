1use std::io;
2
3use fern::colors::{Color, ColoredLevelConfig};
4use log::debug;
5
6use crate::config::logger_config::LoggerConfig;
7use crate::debug_ui;
8use crate::debug_ui::log_ring::{self, RingWriter};
9
10/// Logging utility
11pub(crate) struct Logger;
12
13#[allow(clippy::format_in_format_args)]
14impl Logger {
15    /// This will create and apply a logging dispatcher based upon fern.
16    /// If None is provided, LoggerConfig::Default will be used.
17    /// In every case, will be called at app init, and this will try to
18    /// apply a logging config. If one already exists, it won't replace it.
19    pub fn init_logging(config: Option<LoggerConfig>) {
20        let config = config.unwrap_or_default();
21        let enable_test = debug_ui::enabled();
22        let ring = if enable_test {
23            Some(log_ring::init(4096))
24        } else {
25            None
26        };
27        let color_config = ColoredLevelConfig {
28            error: Color::Red,
29            warn: Color::Yellow,
30            info: Color::Blue,
31            debug: Color::White,
32            trace: Color::BrightWhite,
33        };
34        let mut root = fern::Dispatch::new()
35            .format(|out, message, record| {
36                out.finish(format_args!(
37                    "[{}][{}] : \n    -> {}", record.level(), record.target(), message
38                ))
39            })
40            .level(config.level_filter)
41            .chain(
42                fern::Dispatch::new()
43                    .chain(io::stdout())
44                    .level(config.level_filter)
45                    .level_for("rfvp", config.app_level_filter)
46                    .level_for("wgpu_core", log::LevelFilter::Off)
47                    .level_for("wgpu_hal", log::LevelFilter::Off)
48                    .level_for("naga", log::LevelFilter::Off)
49                    .format(move |out, message, record| {
50                        let color = color_config.get_color(&record.level());
51                        out.finish(format_args!(
52                            "{color}{message}{color_reset}",
53                            color = format!("\x1B[{}m", color.to_fg_str()),
54                            message = message,
55                            color_reset = "\x1B[0m",
56                        ))
57                    }),
58            );
59
60        if let Some(ring) = ring {
61            root = root.chain(
62                fern::Dispatch::new()
63                    .chain(Box::new(RingWriter::new(ring)) as Box<dyn io::Write + Send>)
64                    .level(config.level_filter)
65                    .level_for("rfvp", config.app_level_filter)
66                    .level_for("wgpu_core", log::LevelFilter::Off)
67                    .level_for("wgpu_hal", log::LevelFilter::Off)
68                    .level_for("naga", log::LevelFilter::Off)
69                    .format(|out, message, _record| out.finish(format_args!("{}", message))),
70            );
71        }
72
73        root.apply()
74            .unwrap_or_else(|_| debug!("Logger can be set only one time, skipping."));
75    }
76}
