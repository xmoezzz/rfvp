88    pub fn play(
89        &mut self,
90        slot: i32,
91        repeat: bool,
92        volume: f32,
93        pan: f64,
94        fade_in: kira::Tween,
95    ) -> anyhow::Result<()> {
96        let slot = slot as usize;
97
98        self.se_repeat[slot] = repeat;
99        self.se_pan[slot] = pan;
100        self.se_volumes[slot] = volume;
101        let actual_volume = if self.se_muted[slot] { 0.0 } else { volume };let bgm = match &self.se_datas[slot] {
102            Some(data) => data.clone(),
103            None => {
104                log::error!("Tried to play BGM slot {}, but no BGM was loaded", slot);
105                return Ok(());
106            }
107        };
108
109        log::info!("Playing SE slot {}", slot);
110
111        let loop_region = repeat.then_some(Region::default());
112        let pan = kira::Panning::from(pan as f32);
113        let settings = StaticSoundSettings::new()
114            .panning(pan)
115            .volume(actual_volume)
116            .fade_in_tween(fade_in)
117            .loop_region(loop_region)
118            .playback_rate(1.0);
119
120        let bgm = bgm.with_settings(settings);
121        let handle = self.audio_manager.play(bgm);
122
123        if let Some(mut old_handle) = self.se_slots[slot].take() {
124            old_handle.stop(fade_in);
125        }
126
127        self.se_slots[slot] = Some(handle);
128        Ok(())
129    }

207    pub fn debug_summary(&self) -> SeDebugSummary {
208        let loaded_datas = self.se_datas.iter().filter(|x| x.is_some()).count();
209        let playing_slots = self.se_slots.iter().filter(|x| x.is_some()).count();
210
211        let mut slots = Vec::new();
212        for slot in 0..SE_SLOT_COUNT {
213            let playing = self.se_slots[slot].is_some();
214            let data_loaded = self.se_datas[slot].is_some();
215            let has_name = self.se_names[slot].is_some();
216            let has_kind = self.se_kinds[slot].is_some();
217            if playing || data_loaded || has_name || has_kind {
218                slots.push(SeSlotInfo {
219                    slot,
220                    name: self.se_names[slot].clone(),
221                    volume: self.se_volumes[slot] as f64,
222                    muted: self.se_muted[slot],
223                    kind: self.se_kinds[slot],
224                    data_loaded,
225                    playing,
226                });
227            }
228        }
229
230        SeDebugSummary {
231            max_datas: self.se_datas.len(),
232            max_slots: self.se_slots.len(),
233            loaded_datas,
234            playing_slots,
235            slots,
236        }
237    }

239    pub fn capture_snapshot_v1(&self) -> SePlayerSnapshotV1 {
240        let mut slots: Vec<SeSlotSnapshotV1> = Vec::new();
241
242        for i in 0..SE_SLOT_COUNT {
243            let has_any = self.se_datas[i].is_some() || self.se_slots[i].is_some() || self.se_names[i].is_some();
244            if !has_any {
245                continue;
246            }
247
248            slots.push(SeSlotSnapshotV1 {
249                slot: i as u16,
250                path: self.se_names[i].clone(),
251                sound_type: self.se_kinds[i],
252                volume: self.se_volumes[i] as f32,
253                muted: self.se_muted[i],
254                playing: self.se_slots[i].is_some(),
255                repeat: self.se_repeat[i],
256                pan: self.se_pan[i] as f32,
257            });
258        }
259
260        SePlayerSnapshotV1 { version: 1, slots }
261    }

263    pub fn apply_snapshot_v1(&mut self, snap: &SePlayerSnapshotV1, vfs: &Vfs) -> Result<()> {
264        if snap.version != 1 {
265            return Err(anyhow!("unsupported SePlayerSnapshotV1 version: {}", snap.version));
266        }
267
268        // Stop and clear current state.
269        for i in 0..SE_SLOT_COUNT {
270            self.stop(i as i32, Tween::default());
271            self.se_datas[i] = None;
272            self.se_kinds[i] = None;
273            self.se_names[i] = None;
274            self.se_muted[i] = false;
275            self.se_volumes[i] = 100.0;
276            self.se_repeat[i] = false;
277            self.se_pan[i] = 0.5;
278        }
279
280        for s in &snap.slots {
281            let slot = s.slot as usize;
282            if slot >= SE_SLOT_COUNT {
283                continue;
284            }
285
286            self.se_kinds[slot] = s.sound_type;
287            self.se_muted[slot] = s.muted;
288            self.se_volumes[slot] = s.volume;
289            self.se_repeat[slot] = s.repeat;
290            self.se_pan[slot] = s.pan as f64;
291
292            if let Some(path) = s.path.clone() {
293                let data = vfs
294                    .read_file(&path)
295                    .with_context(|| format!("read SE from vfs: {}", path))?;
296                self.load_named(slot as i32, path, data)?;
297            }
298
299            if s.playing {
300                if self.se_datas[slot].is_some() {
301                    self.play(
302                        slot as i32,
303                        self.se_repeat[slot],
304                        self.se_volumes[slot] as f32,
305                        self.se_pan[slot],
306                        Tween::default(),
307                    )?;
308
309                    if self.se_muted[slot] {
310                        self.silent_on(slot as i32, Tween::default());
311                    }
312                }
313            }
314        }
315
316        Ok(())
317    }
