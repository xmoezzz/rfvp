19    // get pixels as RGBA8 slice
20pub fn map_to_rgba8(&self, device: &wgpu::Device) -> Vec<u8> {
21        let height = self.height as usize;
22        let dst_bpr = self.bytes_per_row as usize;
23        let src_bpr = self.padded_bytes_per_row as usize;
24
25        if height == 0 || dst_bpr == 0 {
26            return Vec::new();
27        }
28
29        let slice = self.buffer.slice(..);
30
31        // wgpu::Buffer::map_async is async; block here without extra crates.
32        let (tx, rx) = std::sync::mpsc::channel();
33        slice.map_async(wgpu::MapMode::Read, move |res| {
34            let _ = tx.send(res);
35        });
36
37        // Ensure the mapping completes.
38        device.poll(wgpu::Maintain::Wait);
39
40        // Propagate mapping failure loudly (you can change to Result if preferred).
41        rx.recv().expect("map_async callback dropped").expect("buffer map failed");
42
43        let mapped = slice.get_mapped_range();
44
45        // Strip per-row padding.
46        let mut out = vec![0u8; height * dst_bpr];
47        for y in 0..height {
48            let src0 = y * src_bpr;
49            let src1 = src0 + dst_bpr;
50            let dst0 = y * dst_bpr;
51            let dst1 = dst0 + dst_bpr;
52            out[dst0..dst1].copy_from_slice(&mapped[src0..src1]);
53        }
54
55        drop(mapped);
56        self.buffer.unmap();
57
58        out
59    }
60}

72impl RenderTarget {
73    pub const FORMAT: wgpu::TextureFormat = wgpu::TextureFormat::Rgba8UnormSrgb;

75    pub fn new(resources: &GpuCommonResources, size: (u32, u32), label: Option<&str>) -> Self {
76        let (w, h) = (size.0.max(1), size.1.max(1));
77        let texture = resources.device.create_texture(&wgpu::TextureDescriptor {
78            label,
79            size: wgpu::Extent3d { width: w, height: h, depth_or_array_layers: 1 },
80            mip_level_count: 1,
81            sample_count: 1,
82            dimension: wgpu::TextureDimension::D2,
83            format: Self::FORMAT,
84            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST | wgpu::TextureUsages::COPY_SRC,
85            view_formats: &[],
86        });
87
88        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
89        let sampler = resources.device.create_sampler(&wgpu::SamplerDescriptor {
90            label: Some("rfvp_render.rendertarget_sampler"),
91            address_mode_u: wgpu::AddressMode::ClampToEdge,
92            address_mode_v: wgpu::AddressMode::ClampToEdge,
93            address_mode_w: wgpu::AddressMode::ClampToEdge,
94            mag_filter: wgpu::FilterMode::Nearest,
95            min_filter: wgpu::FilterMode::Nearest,
96            mipmap_filter: wgpu::FilterMode::Nearest,
97            ..Default::default()
98        });
99
100        let bind_group = resources.device.create_bind_group(&wgpu::BindGroupDescriptor {
101            label: Some("rfvp_render.rendertarget_bind_group"),
102            layout: &resources.bind_group_layouts.texture,
103            entries: &[
104                wgpu::BindGroupEntry { binding: 0, resource: wgpu::BindingResource::TextureView(&view) },
105                wgpu::BindGroupEntry { binding: 1, resource: wgpu::BindingResource::Sampler(&sampler) },
106            ],
107        });
108
109        // Fullscreen quad in virtual pixel space.
110        // Coordinate system: origin at top-left, x right, y down.
111        let vw = w as f32;
112        let vh = h as f32;
113        let x0 = 0.0;
114        let y0 = 0.0;
115        let x1 = vw;
116        let y1 = vh;
117
118        let white = Vec4::ONE;
119        let v = [
120            // Two triangles (0,1,2) (2,1,3)
121            PosColTexVertex { position: glam::vec3(x0, y1, 0.0), color: white, texture_coordinate: glam::vec2(0.0, 1.0) },
122            PosColTexVertex { position: glam::vec3(x0, y0, 0.0), color: white, texture_coordinate: glam::vec2(0.0, 0.0) },
123            PosColTexVertex { position: glam::vec3(x1, y1, 0.0), color: white, texture_coordinate: glam::vec2(1.0, 1.0) },
124            PosColTexVertex { position: glam::vec3(x1, y1, 0.0), color: white, texture_coordinate: glam::vec2(1.0, 1.0) },
125            PosColTexVertex { position: glam::vec3(x0, y0, 0.0), color: white, texture_coordinate: glam::vec2(0.0, 0.0) },
126            PosColTexVertex { position: glam::vec3(x1, y0, 0.0), color: white, texture_coordinate: glam::vec2(1.0, 0.0) },
127        ];
128
129        let quad_vb = resources.device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
130            label: Some("rfvp_render.rendertarget_quad_vb"),
131            contents: bytemuck::cast_slice(&v),
132            usage: wgpu::BufferUsages::VERTEX,
133        });
134
135        Self {
136            texture,
137            view,
138            sampler,
139            bind_group: TextureBindGroup::new(bind_group),
140            size: (w, h),
141            quad_vb,
142            quad_vertices: 0..6,
143        }
144    }

146    pub fn projection_matrix(&self) -> Mat4 {
147        // Virtual space: origin at top-left, x right, y down.
148        // Maps pixel coordinates (0..w, 0..h) into NDC (-1..1, 1..-1).
149        let (w, h) = (self.size.0 as f32, self.size.1 as f32);
150        mat4(
151            vec4(2.0 / w, 0.0, 0.0, 0.0),
152            vec4(0.0, -2.0 / h, 0.0, 0.0),
153            vec4(0.0, 0.0, 1.0, 0.0),
154            vec4(-1.0, 1.0, 0.0, 1.0),
155        )
156    }

192    /// Encode a GPU -> CPU readback of the full render target into an internal buffer.
193    /// The returned buffer is MAP_READ and must be mapped after submission.
194    pub fn encode_readback_rgba8(
195        &self,
196        device: &wgpu::Device,
197        encoder: &mut wgpu::CommandEncoder,
198    ) -> RenderTargetReadback {
199        let width = self.size.0;
200        let height = self.size.1;
201        let bytes_per_row = 4u32.saturating_mul(width);
202        let align = wgpu::COPY_BYTES_PER_ROW_ALIGNMENT;
203        let padded_bytes_per_row = ((bytes_per_row + align - 1) / align) * align;
204
205        let buffer_size = (padded_bytes_per_row as u64) * (height as u64);
206        let buffer = device.create_buffer(&wgpu::BufferDescriptor {
207            label: Some("rfvp_render.render_target.readback"),
208            size: buffer_size,
209            usage: wgpu::BufferUsages::COPY_DST | wgpu::BufferUsages::MAP_READ,
210            mapped_at_creation: false,
211        });
212
213        encoder.copy_texture_to_buffer(
214            wgpu::ImageCopyTexture {
215                texture: &self.texture,
216                mip_level: 0,
217                origin: wgpu::Origin3d::ZERO,
218                aspect: wgpu::TextureAspect::All,
219            },
220            wgpu::ImageCopyBuffer {
221                buffer: &buffer,
222                layout: wgpu::ImageDataLayout {
223                    offset: 0,
224                    bytes_per_row: Some(padded_bytes_per_row),
225                    rows_per_image: Some(height),
226                },
227            },
228            wgpu::Extent3d {
229                width,
230                height,
231                depth_or_array_layers: 1,
232            },
233        );
234
235        RenderTargetReadback {
236            buffer,
237            width,
238            height,
239            bytes_per_row,
240            padded_bytes_per_row,
241        }
242    }
