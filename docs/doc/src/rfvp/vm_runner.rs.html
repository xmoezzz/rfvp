15/// Drives the script VM (which is coroutine-based, not OS-thread based).
16///
17/// Design goal: isolate context-switching and opcode execution from the rest of the engine loop,
18/// while keeping the refactor minimally invasive.
19#[derive(Debug)]
20pub struct VmRunner {
21    tm: ThreadManager,
22}
23

25pub struct VmTickReport {

27    /// With opcode-budget slicing removed, the VM no longer reports artificial forced yields.
28    pub forced_yield: bool,
29    /// Reserved for compatibility with the existing host/worker interface.
30    pub forced_yield_contexts: u32,
31}

50    /// Execute one engine frame worth of script VM work.
51    ///
52    /// `frame_time_ms` is the elapsed time budget for timers (wait/sleep/etc.).
53    pub fn tick(&mut self, game: &mut GameData, parser: &mut Parser, frame_time_ms: u64) -> Result<VmTickReport> {
54        self.process_deferred_text_requests(game);
55        // The VM itself is cooperative; the engine decides when to advance contexts.
56        // If the game is halted (e.g. waiting for IO / modal UI), we do not advance contexts.
57        if game.get_halt() {
58            if debug_ui::enabled() {
59                game.debug_vm_mut().update_from_thread_manager(&self.tm);
60            }
61            return Ok(VmTickReport::default());
62        }
63
64        // If a save capture is pending, snapshot the VM state now (on the VM thread) so the
65// render thread can serialize it without accessing the VM internals.
66if game.save_manager.wants_vm_snapshot_capture() && !game.save_manager.has_pending_vm_snapshot() {
67    let snap = self.tm.capture_snapshot_v1();
68    game.save_manager.set_pending_vm_snapshot(snap);
69}
70
71// Process deferred load requests at a safe point (between VM ticks).
72if let Some(slot) = game.save_manager.take_load_request() {
73    let path = SaveItem::resolve_save_path_for_read(slot);
74    match fs::read(&path) {
75        Ok(bytes) => {
76            let nls = game.get_nls();
77            if let Err(e) = game.save_manager.load_slot_into_current_from_bytes(slot, nls, &bytes) {
78                log::error!("load: failed to parse save header for slot {}: {:#}", slot, e);
79            }
80
81            match try_decode_state_chunk_v1(&bytes) {
82                Ok(Some(s)) => {
83                    if let Err(e) = s.apply(game, &mut self.tm) {
84                        log::error!("load: apply SaveStateSnapshotV1 failed: {:#}", e);
85                    }
86                }
87                Ok(None) => {
88                    // No RFVS chunk: header-only save (engine save or older rfvp save).
89                    log::warn!("load: no RFVS chunk found in slot {} (header-only load)", slot);
90                }
91                Err(e) => {
92                    log::error!("load: failed to decode RFVS chunk for slot {}: {:#}", slot, e);
93                }
94            }
95        }
96        Err(e) => {
97            log::error!("load: failed to read save slot {} from {}: {:#}", slot, path.display(), e);
98        }
99    }
100
101    // Do not advance contexts in the same tick; resume on the next frame.
102    if debug_ui::enabled() {
103        game.debug_vm_mut().update_from_thread_manager(&self.tm);
104    }
105    return Ok(VmTickReport::default());
106}
107
108
109
110// In the original engine, dissolve is a global visual state that can unblock VM waits.
111        let dissolve_type = game.motion_manager.get_dissolve_type();
112        let dissolve2_transitioning = game.motion_manager.is_dissolve2_transitioning();
113
114        let report = VmTickReport::default();
115
116        let total = self.tm.total_contexts() as u32;
117        for tid in 0..total {
118            if game.get_halt() {
119                break;
120            }
121
122            // Legacy behavior: when the game is terminating, only the last-running context keeps advancing.
123            if game.get_game_should_exit() && game.get_last_current_thread() != tid {
124                continue;
125            }
126
127            self.advance_timers_and_state(tid, dissolve_type, dissolve2_transitioning, frame_time_ms);
128
129            let status = self.tm.get_context_status(tid);
130            if status.contains(ThreadState::CONTEXT_STATUS_RUNNING)
131                && !status.contains(ThreadState::CONTEXT_STATUS_WAIT)
132                && !status.contains(ThreadState::CONTEXT_STATUS_SLEEP)
133                && !status.contains(ThreadState::CONTEXT_STATUS_TEXT)
134                && !status.contains(ThreadState::CONTEXT_STATUS_DISSOLVE_WAIT)
135            {
136                self.run_one_context(tid, game, parser)?;
137            }
138        }
139
140        // ExitMode(3): once the designated "last current" context has actually exited,
141        // signal the host loop. Ordinary should_break/yield must not count as exit.
142        if game.get_lock_scripter() {
143            let main_tid = game.get_last_current_thread();
144            let st = self.tm.get_context_status(main_tid);
145            if st == ThreadState::CONTEXT_STATUS_NONE {
146                game.set_main_thread_exited(true);
147            }
148        }
149
150
151// If a save capture is pending, snapshot VM state after the tick so the render thread
152// can serialize a consistent coroutine state for this frame.
153if game.save_manager.wants_vm_snapshot_capture() {
154    let snap = self.tm.capture_snapshot_v1();
155    game.save_manager.set_pending_vm_snapshot(snap);
156}
157
158        if debug_ui::enabled() {
159            game.debug_vm_mut().update_from_thread_manager(&self.tm);
160        }
161
162        Ok(report)
163    }

165    fn process_deferred_text_requests(&mut self, game: &mut GameData) {
166        let mut keep = std::collections::VecDeque::new();
167        while let Some(req) = game.thread_wrapper.pop() {
168            match req {
169                ThreadRequest::TextResume(id) => {
170                    let mut st = self.tm.get_context_status(id);
171                    st.remove(ThreadState::CONTEXT_STATUS_TEXT);
172                    st.insert(ThreadState::CONTEXT_STATUS_RUNNING);
173                    self.tm.set_context_status(id, st);
174                }
175                other => keep.push_back(other),
176            }
177        }
178        while let Some(req) = keep.pop_front() {
179            match req {
180                ThreadRequest::Start(id, addr) => game.thread_wrapper.thread_start(id, addr),
181                ThreadRequest::Wait(time) => game.thread_wrapper.thread_wait(time),
182                ThreadRequest::DissolveWait() => game.thread_wrapper.dissolve_wait(),
183                ThreadRequest::Sleep(time) => game.thread_wrapper.thread_sleep(time),
184                ThreadRequest::Raise(time) => game.thread_wrapper.thread_raise(time),
185                ThreadRequest::Next() => game.thread_wrapper.thread_next(),
186                ThreadRequest::TextWait(id) => game.thread_wrapper.thread_text_wait(id),
187                ThreadRequest::Exit(id) => game.thread_wrapper.thread_exit(id),
188                ThreadRequest::ShouldBreak() => game.thread_wrapper.should_break(),
189                ThreadRequest::TextResume(id) => game.thread_wrapper.thread_text_resume(id),
190            }
191        }
192    }

194    fn advance_timers_and_state(&mut self, tid: u32, dissolve_type: DissolveType, dissolve2_transitioning: bool, frame_time_ms: u64) {
195        let status = self.tm.get_context_status(tid);
196
197        // WAIT timer
198        if status.contains(ThreadState::CONTEXT_STATUS_WAIT) {
199            let wait_time = self.tm.get_context_waiting_time(tid);
200            if wait_time > frame_time_ms {
201                self.tm
202                    .set_context_waiting_time(tid, wait_time - frame_time_ms);
203            } else {
204                self.tm.set_context_waiting_time(tid, 0);
205                let mut new_status = status.clone();
206                new_status.remove(ThreadState::CONTEXT_STATUS_WAIT);
207                // Resume execution once WAIT expires.
208                new_status.insert(ThreadState::CONTEXT_STATUS_RUNNING);
209                self.tm.set_context_status(tid, new_status);
210            }
211        }
212
213        // SLEEP timer
214        if status.contains(ThreadState::CONTEXT_STATUS_SLEEP) {
215            let sleep_time = self.tm.get_context_sleeping_time(tid);
216            if sleep_time > frame_time_ms {
217                self.tm.set_context_sleeping_time(tid, sleep_time - frame_time_ms);
218            } else {
219                self.tm.set_context_sleeping_time(tid, 0);
220                let mut new_status = status.clone();
221                new_status.remove(ThreadState::CONTEXT_STATUS_SLEEP);
222                new_status.insert(ThreadState::CONTEXT_STATUS_RUNNING);
223                self.tm.set_context_status(tid, new_status);
224            }
225        }
226
227        // Dissolve wait is unblocked when dissolve is completed / static, and dissolve2 is not transitioning.
228        if status.contains(ThreadState::CONTEXT_STATUS_DISSOLVE_WAIT)
229            && (dissolve_type == DissolveType::None || dissolve_type == DissolveType::Static)
230            && !dissolve2_transitioning
231        {
232            let mut new_status = status.clone();
233            new_status.remove(ThreadState::CONTEXT_STATUS_DISSOLVE_WAIT);
234            // Resume execution once dissolve completes/static.
235            new_status.insert(ThreadState::CONTEXT_STATUS_RUNNING);
236            self.tm.set_context_status(tid, new_status);
237        }
238    }

240    fn run_one_context(
241        &mut self,
242        tid: u32,
243        game: &mut GameData,
244        parser: &mut Parser,
245    ) -> Result<()> {
246        // Keep the VM's notion of current thread aligned with GameData.
247        self.tm.set_current_id(tid);
248        game.set_current_thread(tid);
249
250        self.tm.set_context_should_break(tid, false);
251        while !self.tm.get_context_should_break(tid) {
252            let result = self.tm.context_dispatch_opcode(tid, game, parser);
253
254            if self.tm.get_contexct_should_exit(tid) {
255                self.tm.thread_exit(Some(tid));
256                break;
257            }
258
259            if let Err(e) = result {
260                // Preserve the previous "fail fast" behavior for now.
261                log::error!("Error while executing the script: {:#?}", e);
262                anyhow::bail!(e);
263            }
264
265            let mut must_yield = false;
266
267            // Drain all pending requests emitted by syscalls.
268            while let Some(event) = game.thread_wrapper.pop() {
269                match event {
270                    ThreadRequest::Start(id, addr) => self.tm.thread_start(id, addr),
271                    ThreadRequest::Wait(time) => {
272                        self.tm.thread_wait(time);
273                        must_yield = true;
274                    },
275                    ThreadRequest::DissolveWait() => {
276                        self.tm.thread_dissolve_wait();
277                        must_yield = true;
278                    },
279                    ThreadRequest::Sleep(time) => {
280                        self.tm.thread_sleep(time);
281                        must_yield = true;
282                    }
283                    ThreadRequest::Raise(time) => {
284                        self.tm.thread_raise(time);
285                        must_yield = true;
286                    }
287                    ThreadRequest::Next() => {
288                        self.tm.thread_next();
289                        must_yield = true;
290                    }
291                    ThreadRequest::TextWait(id) => {
292                        let mut st = self.tm.get_context_status(id);
293                        st.insert(ThreadState::CONTEXT_STATUS_TEXT);
294                        st.remove(ThreadState::CONTEXT_STATUS_RUNNING);
295                        self.tm.set_context_status(id, st);
296                        must_yield = true;
297                    }
298                    ThreadRequest::TextResume(id) => {
299                        let mut st = self.tm.get_context_status(id);
300                        st.remove(ThreadState::CONTEXT_STATUS_TEXT);
301                        st.insert(ThreadState::CONTEXT_STATUS_RUNNING);
302                        self.tm.set_context_status(id, st);
303                    }
304                    ThreadRequest::Exit(id) => {
305                        self.tm.thread_exit(id);
306                        must_yield = true;
307                    }
308                    ThreadRequest::ShouldBreak() => {
309                        // Must break the CURRENT context, not a global flag.
310                        self.tm.set_context_should_break(tid, true);
311                        self.tm.set_should_break(true);
312                        must_yield = true;
313                    }
314                }
315            }
316
317            if must_yield {
318                // Force a per-context yield at frame boundary.
319                self.tm.set_context_should_break(tid, true);
320                break;
321            }
322        }
323
324        Ok(())
325    }
