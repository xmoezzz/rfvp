11#[derive(Debug, Clone, Copy, Default)]
12pub enum Nls {
13    #[default]
14    ShiftJIS = 0,
15    GBK = 1,
16    UTF8 = 2,
17}
18
19
20impl FromStr for Nls {
21    type Err = anyhow::Error;
22
23    fn from_str(s: &str) -> anyhow::Result<Self> {
24        match s.to_ascii_lowercase().as_str() {
25            "sjis" => Ok(Nls::ShiftJIS),
26            "gbk" => Ok(Nls::GBK),
27            "utf8" => Ok(Nls::UTF8),
28            _ => Err(anyhow::anyhow!("unknown NLS")),
29        }
30    }
31}

86    /// safely read a u8 from the buffer
87    pub fn read_u8(&self, offset: usize) -> Result<u8> {
88        if offset >= self.buffer.len() {
89            return Err(anyhow::anyhow!("offset out of bounds"));
90        }
91        Ok(self.buffer[offset])
92    }
93
94    /// safely read a little-endian u16 from the buffer
95    pub fn read_u16(&self, offset: usize) -> Result<u16> {
96        if offset + 1 >= self.buffer.len() {
97            return Err(anyhow::anyhow!("offset out of bounds"));
98        }
99        Ok(u16::from_le_bytes([self.buffer[offset], self.buffer[offset + 1]]))
100    }
101
102    /// safely read a little-endian u32 from the buffer
103    pub fn read_u32(&self, offset: usize) -> Result<u32> {
104        if offset + 3 >= self.buffer.len() {
105            return Err(anyhow::anyhow!("offset out of bounds"));
106        }
107        Ok(u32::from_le_bytes([
108            self.buffer[offset],
109            self.buffer[offset + 1],
110            self.buffer[offset + 2],
111            self.buffer[offset + 3],
112        ]))
113    }

115    /// safely read a little-endian i8 from the buffer

117        if offset >= self.buffer.len() {

157    /// safe read a c-style string from the buffer with string length
158    /// (with null terminator)
159    /// then convert it to a UTF-8 string due to the NLS
160    pub fn read_cstring(&self, offset: usize, len: usize) -> Result<String> {
161        if offset + len >= self.buffer.len() {
162            return Err(anyhow::anyhow!("offset out of bounds"));
163        }
164        let mut string = Vec::new();
165        for i in 0..len {
166            if self.buffer[offset + i] == 0 {
167                break;
168            }
169            string.push(self.buffer[offset + i]);
170        }
171
172        if string.ends_with(&[0]) {
173            string.pop();
174        }
175        
176        let s = match self.nls {
177            Nls::ShiftJIS => {
178                let (s, _, e) = encoding_rs::SHIFT_JIS.decode(&string);
179                if e {
180                    log::error!("failed to decode string as ShiftJIS");
181                }
182                s
183            }
184            Nls::GBK => {
185                let (s, _, e) = encoding_rs::GBK.decode(&string);
186                if e {
187                    log::error!("failed to decode string as GBK");
188                }
189                s
190            }
191            Nls::UTF8 => {
192                let (s, _, e) = encoding_rs::UTF_8.decode(&string);
193                if e {
194                    log::error!("failed to decode string as UTF-8");
195                }
196                s
197            }
198        };
199
200        Ok(s.to_string())
201    }

203    fn parser(&mut self) -> Result<()> {
204        let mut off = 0usize;
205        self.sys_desc_offset = self.read_u32(off)?;
206
207        off = self.sys_desc_offset as usize;
208        self.entry_point = self.read_u32(off)?;
209        off += size_of::<u32>();
210
211        self.non_volatile_global_count = self.read_u16(off)?;
212        off += size_of::<u16>();
213
214        self.volatile_global_count = self.read_u16(off)?;
215        off += size_of::<u16>();
216
217        self.game_mode = self.read_u8(off)? as u8;
218        off += size_of::<u8>();
219
220        self.game_mode_reserved = self.read_u8(off)? as u8;
221        off += size_of::<u8>();
222
223        let title_len = self.read_u8(off)?;
224        off += size_of::<u8>();
225
226        self.game_title = self.read_cstring(off, title_len as usize)?;
227        off += title_len as usize;
228
229        self.syscall_count = self.read_u16(off)?;
230        off += size_of::<u16>();
231
232        for i in 0..self.syscall_count {
233            let args = self.read_u8(off)?;
234            off += size_of::<u8>();
235
236            let name_len = self.read_u8(off)?;
237            off += size_of::<u8>();
238
239            let name = self.read_cstring(off, name_len as usize)?;
240            off += name_len as usize;
241
242            self.syscalls.insert(i as usize, Syscall { args, name });
243        }
244
245        self.custom_syscall_count = self.read_u16(off)?;
246        if self.custom_syscall_count > 0 {
247            log::warn!("custom syscall count: {}", self.custom_syscall_count);
248        }
249
250        Ok(())
251    }
