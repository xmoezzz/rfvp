168    /// # Safety

171        use std::ptr;

177        ptr::addr_of_mut!((*dst).history_manager).write(HistoryManager::default());
178        ptr::addr_of_mut!((*dst).flag_manager).write(FlagManager::default());
179        ptr::addr_of_mut!((*dst).motion_manager).write(MotionManager::default());

183        ptr::addr_of_mut!((*dst).video_manager).write(VideoPlayerManager::default());
184        ptr::addr_of_mut!((*dst).save_manager).write(SaveManager::default());
185        ptr::addr_of_mut!((*dst).nls).write(Nls::default());
186        ptr::addr_of_mut!((*dst).memory_cache).write(Vec::new());
187        ptr::addr_of_mut!((*dst).time).write(Time::default());

191        ptr::addr_of_mut!((*dst).bgm_player).write(BgmPlayer::new(audio_manager.clone()));
192        ptr::addr_of_mut!((*dst).audio_manager).write(audio_manager);
193        ptr::addr_of_mut!((*dst).root_prim).write(None);

196        ptr::addr_of_mut!((*dst).can_fullscreen).write(false);

202        ptr::addr_of_mut!((*dst).close_pending).write(false);

211    }

498impl VmSyscall for GameData {
499    fn do_syscall(&mut self, name: &str, args: Vec<Variant>) -> anyhow::Result<Variant> {
500        let result = SYSCALL_TBL.borrow().get(name).map_or_else(
501            || {
502                log::error!("Syscall {} not found", name);
503                Ok(Variant::Nil)
504            },
505            |syscall| syscall.call(self, args),
506        );
507        result
508    }
509}
