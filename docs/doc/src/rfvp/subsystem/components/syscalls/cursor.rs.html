9/// CursorShow(show: int)
10/// IDA SYSCALL_SPECS: argc=1
11pub struct CursorShow;
12impl Syscaller for CursorShow {
13    fn call(&self, game_data: &mut GameData, args: Vec<Variant>) -> Result<Variant> {
14        let show = !get_var!(args, 0).is_nil();
15        // Cursor visibility is handled at the window layer.
16        // Keep the value for engine logic; the platform backend can consume it.
17        game_data.window_mut().set_cursor_visible(show);
18        Ok(Variant::Nil)
19    }
20}

22/// CursorMove(x: int, y: int, force: any)
23/// IDA SYSCALL_SPECS: argc=3
24///
25/// Original engine behavior (IDA decompilation):
26/// - Only moves when args[0] and args[1] are integers.
27/// - Movement is enabled when either:
28///   (a) global int var (non_volatile_global_count + 15) equals 1, or
29///   (b) args[2] is non-nil (force move).
30pub struct CursorMove;
31impl Syscaller for CursorMove {
32    fn call(&self, game_data: &mut GameData, args: Vec<Variant>) -> Result<Variant> {
33        let Some(x) = get_var!(args, 0).as_int() else {
34            return Ok(Variant::Nil);
35        };
36        let Some(y) = get_var!(args, 1).as_int() else {
37            return Ok(Variant::Nil);
38        };
39
40        let force = !get_var!(args, 2).is_nil();
41        let allow = get_int_var(15) == 1;
42
43        if allow || force {
44            game_data.inputs_manager.notify_mouse_move(x, y);
45            game_data.inputs_manager.set_mouse_in(true);
46            game_data.window_mut().set_cursor_pos(x, y);
47        }
48
49        Ok(Variant::Nil)
50    }
51}

53/// CursorChange(id: int)
54/// IDA SYSCALL_SPECS: argc=1
55pub struct CursorChange;
56impl Syscaller for CursorChange {
57    fn call(&self, game_data: &mut GameData, args: Vec<Variant>) -> Result<Variant> {
58        let Some(id) = get_var!(args, 0).as_int() else {
59            return Ok(Variant::Nil);
60        };
61        if !(0..4).contains(&id) {
62            return Ok(Variant::Nil);
63        }
64        // Engine cursor shape/theme. If your backend supports it, map id to a winit CursorIcon.
65        game_data.window_mut().set_cursor_kind(id);
66        Ok(Variant::Nil)
67    }
68}
