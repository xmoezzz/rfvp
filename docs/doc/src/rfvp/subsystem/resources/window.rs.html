1use winit::window::CursorIcon;

3#[derive(Default, Debug, Copy, Clone)]
4struct FutureSettings{
5    new_cursor: Option<CursorIcon>,
6    dimensions: Option<(u32, u32)>,
7    cursor_visible: Option<bool>,
8    cursor_pos: Option<(i32, i32)>,
9}
10/// [`Window`] is a Resource dedicated to have an access control over the current window.
11/// Its size is immediately updated when the window resize event happens.
12/// new_cursor is set at the end of the current frame.
13#[derive(Default, Debug, Copy, Clone)]
14pub struct Window {
15    width: u32,
16    height: u32,
17    dpi: f64,
18    future_settings: FutureSettings
19}
20
21impl Window {
22    pub(crate) fn new(screen_size: (u32, u32), dpi: f64) -> Self {
23        Self { width: screen_size.0, height: screen_size.1, future_settings: Default::default(), dpi }
24    }
25
26    pub(crate) fn set_dimensions(&mut self, width: u32, height: u32) {
27        self.width = width;
28        self.height = height;
29    }
30
31    pub fn set_cursor(&mut self, icon: CursorIcon) {
32        self.future_settings.new_cursor = Some(icon);
33    }
34
35    pub fn resize(&mut self, width: u32, height: u32) {
36        self.future_settings.dimensions = Some((width, height));
37    }
38
39    pub(crate) fn reset_future_settings(&mut self) {
40        self.future_settings = Default::default();
41    }
42
43    pub fn dimensions(&self) -> (u32, u32) {
44        (self.width, self.height)
45    }
46
47    pub fn width(&self) -> u32 {
48        self.width / self.dpi as u32
49    }
50
51    pub fn height(&self) -> u32 {
52        self.height / self.dpi as u32
53    }
54
55    pub fn dpi(&self) -> f64 {
56        self.dpi
57    }
58
59    pub fn new_cursor(&self) -> &Option<CursorIcon> {
60        &self.future_settings.new_cursor
61    }
62    pub fn new_dimensions(&self) -> &Option<(u32, u32)> {
63        &self.future_settings.dimensions
64    }
65
66    pub fn new_cursor_visible(&self) -> Option<bool> {
67        self.future_settings.cursor_visible
68    }
69
70    pub fn new_cursor_pos(&self) -> Option<(i32, i32)> {
71        self.future_settings.cursor_pos
72    }
73    
74    // --- Cursor control (used by Cursor* syscalls) ---
75    pub fn set_cursor_visible(&mut self, visible: bool) {
76        self.future_settings.cursor_visible = Some(visible);
77    }
78
79    pub fn set_cursor_pos(&mut self, x: i32, y: i32) {
80        self.future_settings.cursor_pos = Some((x, y));
81    }
82
83    pub fn set_cursor_kind(&mut self, _kind_id: i32) {
84        // Minimal mapping: treat any id as default cursor for now.
85        self.future_settings.new_cursor = Some(CursorIcon::Default);
86    }
