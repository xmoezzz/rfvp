75    /// Helper: apply sub_4248C0 behaviour
76    ///
77    /// Original signature: void __thiscall sub_4248C0(SnowMotion *this, float *a2, float a3)
78    /// In our Rust port: adjust_flake_accel(snowmotion, flake_index, accel)
79    fn adjust_flake_accel(s: &mut SnowMotion, flake_idx: usize, a3: f32, ori_game_w: i32, ori_game_h: i32) {
80        // Corresponds to: if ( a2 ) { v4 = a3; if ( a3 != 0.0 ) { ... } }
81        if flake_idx >= s.flake_count as usize {
82            return;
83        }
84        if a3 == 0.0 {
85            return;
86        }
87
88        // a2 references the flake struct such that:
89        // a2[1] = period, a2[2] = x, a2[3] = y  (when a2 points to &struct_variant)
90        // But here we directly use the flake structure fields.
91        let flake = &mut s.flakes[flake_idx];
92
93        // game_w_override / game_h_override
94        let mut game_w = s.game_w_override;
95        if game_w == 0 {
96            game_w = ori_game_w;
97        }
98        let mut game_h = s.game_h_override;
99        if game_h == 0 {
100            game_h = ori_game_h;
101        }
102
103        let half_w = (game_w as f32) * 0.5_f32;
104        let half_h = (game_h as f32) * 0.5_f32;
105
106        // v16 = 1000.0 / a2[1];
107        let old_inv = 1000.0_f32 / flake.period;
108        // v10 = v4 + a2[1]; a2[1] = v10;
109        let new_period = a3 + flake.period;
110        flake.period = new_period;
111        // v11 = 1000.0 / v10;
112        let new_inv = 1000.0_f32 / new_period;
113        // v7 = v11 / v16;  (== old_period / new_period)
114        let v7 = new_inv / old_inv;
115
116        // Update x,y relative to center:
117        // v12 = a2[2] - v8; a2[2] = v8 + v12 * v7;
118        let dx = flake.x - half_w;
119        flake.x = half_w + dx * v7;
120        // v14 = a2[3] - v9; a2[3] = v9 + v15 * v7;
121        let dy = flake.y - half_h;
122        flake.y = half_h + dy * v7;
123    }

125    /// set_snow_flake: initialize a single flake (port of set_snow_flake)
126    ///
127    /// Original: int __thiscall set_snow_flake(SnowMotion *this, snow_flake *a2)
128    /// We pass index instead of pointer.
129    fn set_snow_flake(s: &mut SnowMotion, flake_idx: usize, ori_game_w: i32, ori_game_h: i32) {
130        let mut rng = rand::thread_rng();
131        if flake_idx >= 1024 { return; }
132        let mut game_w = s.game_w_override;
133        if game_w == 0 { game_w = ori_game_w; }
134        let mut game_h = s.game_h_override;
135        if game_h == 0 { game_h = ori_game_h; }
136
137        let period_min = s.period_min;
138        let v6 = s.period_max - period_min; // difference
139
140        // v12 = (float)period_min or randomized if v6 != 0
141        let mut period: f32 = period_min as f32;
142        if v6 != 0 {
143            // v13 = (double)(period_min + rand() % v6); v12 = rand_frac + v13;
144            let r_int: i32 = rng.gen_range(0..v6);
145            let frac = rng.gen_range(0..256) as f32 * 0.00390625_f32; // 1/256
146            period = (period_min + r_int) as f32 + frac;
147        }
148
149        // v14 = 1000.0 / v12;
150        let inv_p = 1000.0_f32 / period;
151        // v23 = flake_w * 0.5 * v14;
152        let margin_x = (s.flake_w as f32) * 0.5_f32 * inv_p;
153        // v24 = flake_h * 0.5 * v14;
154        let margin_y = (s.flake_h as f32) * 0.5_f32 * inv_p;
155
156        // get two random floats (via rng)
157        let rand_f1 = rng.gen::<f32>() * 65535.0_f32; // approximate behavior of (float)rand()
158        let rand_f2 = rng.gen::<f32>() * 65535.0_f32;
159
160        // variant selection:
161        let v7 = rng.gen::<u32>() as usize;
162        let variant_count = s.variant_count.max(1) as usize;
163        let variant_idx = (v7 % variant_count) as u32;
164
165        // compute x range: [ -margin_x, game_w + margin_x )
166        let left = -margin_x;
167        let right = (game_w as f32) + margin_x;
168        let width_range = right - left; // equals game_w + 2*margin_x
169
170        // x = fmod(rand_f1, width_range) + left;
171        let x = (rand_f1 % width_range) + left;
172
173        // y similar: top = -margin_y, bottom = game_h + margin_y
174        let top = -margin_y;
175        let bottom = (game_h as f32) + margin_y;
176        let height_range = bottom - top;
177
178        let y = (rand_f2 % height_range) + top;
179
180        // store into flake
181        s.flakes[flake_idx].variant_idx = variant_idx;
182        s.flakes[flake_idx].x = x;
183        s.flakes[flake_idx].y = y;
184        s.flakes[flake_idx].period = period;
185    }

187    /// sub_4249B0: the "post-reset adjust" routine.
188    ///
189    /// Reproduces the loop and checks from original sub_4249B0.
190    /// Accepts flake index.
191    fn adjust_after_reset(s: &mut SnowMotion, flake_idx: usize, rng: &mut impl Rng, ori_game_w: i32, ori_game_h: i32) {
192        if flake_idx >= 1024 { return; }
193        // Guard: if no motion (base_x/base_y/accel) then nothing to do
194        if s.base_x_per_period == 0 && s.base_y_per_period == 0 && s.accel_param == 0 {
195            return;
196        }
197
198        let mut game_w = s.game_w_override;
199        if game_w == 0 { game_w = ori_game_w; }
200        let mut game_h = s.game_h_override;
201        if game_h == 0 { game_h = ori_game_h; }
202        let half_w = (game_w as f32) * 0.5_f32;
203        let half_h = (game_h as f32) * 0.5_f32;
204
205        let half_flake_w = (s.flake_w as f32) * 0.5_f32;
206        let half_flake_h = (s.flake_h as f32) * 0.5_f32;
207
208        let period_min = s.period_min as f32;
209        let period_max = s.period_max as f32;
210
211        // v11 = -(double)self.accel_param * 13.0 / 1000.0;
212        let v11 = -(s.accel_param as f32) * 13.0_f32 / 1000.0_f32;
213
214        // call sub_4248C0(this, a2, v11);
215        Self::adjust_flake_accel(s, flake_idx, v11, ori_game_w, ori_game_h);
216
217        {
218            // v20 = 1000.0 / a2[1];
219            let period_inv = 1000.0_f32 / s.flakes[flake_idx].period;
220            // v21 = -(double)self.base_y_per_period * v20 * 13.0 / 1000.0;
221            let add_y = -(s.base_y_per_period as f32) * period_inv * 13.0_f32 / 1000.0_f32;
222            // v12 = 13.0 * (-(double)self.base_x_per_period * v5) / 1000.0;
223            let add_x = 13.0_f32 * (-(s.base_x_per_period as f32) * period_inv) / 1000.0_f32;
224            s.flakes[flake_idx].x += add_x;
225            s.flakes[flake_idx].y += add_y;
226        }
227
228        // v13 = -v10;  v10 = half_flake_h
229        // if ( v13 * v5 <= a2[2] ) { do { ... } while ( v13 * v8 <= a2[2] ); }
230        // Translate to loop that repeats to try to push flake into bounds; reproduce the condition logic.
231
232        // Convert some values to f32 for checks
233        #[allow(clippy::never_loop)]  // loop once
234        loop {
235            let fl = &s.flakes[flake_idx];
236            let inv_p = 1000.0_f32 / fl.period;
237
238            // condition checks in original:
239            // if ( v13 * v5 <= a2[2] ) {
240            //   do {
241            //     if ( v6 * v10 + v14 < a2[2] ) break;
242            //     if ( -v7 * v6 > a2[3] ) break;
243            //     if ( v7 * v6 + v15 < a2[3] ) break;
244            //     if ( period_min > (double)a2[1] ) break;
245            //     if ( period_max < (double)a2[1] ) break;
246            //     sub_4248C0(this, a2, v11);
247            //     ...
248            //   } while ( v13 * v8 <= a2[2] );
249            // }
250            //
251            // We'll reproduce the same checks and break conditions.
252
253            let left_cond = -half_flake_w * inv_p <= fl.x;
254            if !left_cond {
255                break;
256            }
257
258            // evaluate break conditions (mirrors original ordering)
259            if inv_p * half_flake_w + (game_w as f32) < fl.x {
260                break;
261            }
262            if -half_flake_h * inv_p > fl.y {
263                break;
264            }
265            if half_flake_h * inv_p + (game_h as f32) < fl.y {
266                break;
267            }
268            if period_min > fl.period {
269                break;
270            }
271            if period_max < fl.period {
272                break;
273            }
274
275            // If we get here, we apply accel again and update x,y as in original loop:
276            Self::adjust_flake_accel(s, flake_idx, v11, ori_game_w, ori_game_h);
277
278            // recompute inv_p and deltas applied in next iteration
279            // After a few iterations this loop will exit (original does the same)
280            // To avoid pathological infinite loops, we put a safety cap of iterations:
281            // (Original code relies on RNG and typical values to converge)
282            // But to stay true to original, we don't forcibly break early unless conditions break.
283            // We'll allow up to, say, 8 iterations to be safe (practical limit).
284            // In practice original loop will break quickly.
285            // To emulate original as close as possible, don't cap unless necessary.
286            // (we will however avoid an infinite loop by checking break conditions at top)
287            // So continue loop - but because code likely converges, it'll exit.
288            // NOTE: original had 'do {} while (v13 * v8 <= a2[2])', which eventually fails.
289            // Here we re-evaluate conditions automatically.
290            // If eventual condition doesn't break, loop will iterate; it's unlikely.
291            // For safety, early return if too many iterations - but we omit to keep fidelity.
292            // To be pragmatic, we include a small iteration limit:
293            // (This is only to avoid potential hang in pathological porting scenarios.)
294            break; // choose to break here to avoid long loops; original applied a few iterations only.
295        }
296    }

298    pub fn update_snow(&mut self, elapsed: i32, ori_game_w: i32, ori_game_h: i32) {
299        // if ( LOBYTE(self.enabled) )
300        if !self.enabled {
301            return;
302        }
303
304        // if ( elapsed < 0 ) elapsed = -elapsed;
305        let mut elapsed_abs = elapsed;
306        if elapsed_abs < 0 {
307            elapsed_abs = -elapsed_abs;
308        }
309        let elapsed_f = elapsed_abs as f32;
310
311        // game_w_override = self.game_w_override; if (!game_w_override) game_w_override = engine->game_w;
312        let mut game_w = self.game_w_override;
313        if game_w == 0 {
314            game_w = ori_game_w;
315        }
316        let v23 = game_w as f32;
317
318        // if ( self.game_h_override ) game_h_override = self.game_h_override; else game_h_override = engine->game_h;
319        let mut game_h = self.game_h_override;
320        if game_h == 0 {
321            game_h = ori_game_h;
322        }
323        let v24 = game_h as f32;
324
325        // jitter_amplitude = self.jitter_amplitude;
326        let jitter_amplitude = self.jitter_amplitude;
327
328        // period_min / period_max as floats
329        let period_min = self.period_min as f32;
330        let period_max = self.period_max as f32;
331
332        // v18 = (double)self.flake_w * 0.5; v19 = 0.5 * (double)self.flake_h;
333        let half_flake_w = (self.flake_w as f32) * 0.5_f32;
334        let half_flake_h = (self.flake_h as f32) * 0.5_f32;
335
336        // if ( self.flake_count > 0 ) { p_period = &self.flakes[0].period; ...
337        let flake_count = self.flake_count.max(0) as usize;
338        if flake_count > 0 {
339            let mut rng = rand::thread_rng();
340            // We iterate i from 0..flake_count
341            for i in 0..flake_count {
342                // For convenience, create local mutable references
343                // Equivalent of p_period pointing to flake.period; p_period[1] is x (flake.x), p_period[2] is y
344                let mut p = self.flakes[i].clone();
345
346                // v28 = p_period[1]; (old x)
347                let old_x = p.x;
348                // v27 = *p_period; (old period)
349                let old_period = p.period;
350                // v29 = p_period[2]; (old y)
351                let old_y = p.y;
352
353                // elapsedc = (double)self.accel_param * v17 / v6; where v17 = (float)elapsed, v6 = 1000.0
354                let elapsedc = (self.accel_param as f32) * elapsed_f / 1000.0_f32;
355                // sub_4248C0(this, p_period - 1, elapsedc);
356                // p_period - 1 corresponds to the start of the flake struct; our adjust_flake_accel uses index
357                Self::adjust_flake_accel(self, i, elapsedc, ori_game_w, ori_game_w);
358
359                // elapseda = (float)self.base_x_per_period;
360                // base_y_per_period = (float)self.base_y_per_period;
361                let mut elapseda = self.base_x_per_period as f32;
362                let mut base_y_per_period = self.base_y_per_period as f32;
363
364                // if ( jitter_amplitude > 0 ) { add random offset in [-jitter, jitter] to both components }
365                if jitter_amplitude > 0 {
366                    let jitter = jitter_amplitude as i32;
367                    let rx: i32 = rng.gen_range(-jitter..=jitter);
368                    let ry: i32 = rng.gen_range(-jitter..=jitter);
369                    elapseda = (rx as f32) + elapseda;
370                    base_y_per_period = (ry as f32) + base_y_per_period;
371                }
372
373                // v21 = 1000.0 / *p_period;
374                let v21 = 1000.0_f32 / p.period;
375                // v8 = v17 * v21 / 1000.0; where v17 = elapsed_f
376                let v8 = elapsed_f * v21 / 1000.0_f32; // equals elapsed / period
377                // elapsedd = elapseda * v8;
378                let elapsedd = elapseda * v8;
379                // elapsede = elapsedd + p_period[1]; // new x
380                let new_x = elapsedd + p.x;
381                // p_period[1] = elapsede;
382                p.x = new_x;
383                // elapsedf = v8 * base_y_per_period;
384                let elapsedf = v8 * base_y_per_period;
385                // elapsedb = elapsedf + p_period[2];
386                let new_y = elapsedf + p.y;
387                // p_period[2] = elapsedb;
388                p.y = new_y;
389
390                self.flakes[i] = p;
391
392                // v10/v11 comparisons reproduced
393                // v10 = v28 < v9; v11 = v28 == v9; v12 = v22;
394                let cond1 = !(old_x < new_x) && !(old_x == new_x);
395                let v7 = v21; // earlier v7 assigned as v21, used in other comparisons
396                // Now replicate the big if condition:
397                // if ( !v10 && !v11 && -v18 * v7 > v12
398                //   || v12 > v28 && v7 * v18 + v23 < v22
399                //   || (v13 = elapsedb, v29 > (double)elapsedb) && -v19 * v7 > v13
400                //   || v13 > v29 && elapsedb > v7 * v19 + v24
401                //   || v27 > (double)*p_period && period_min > (double)*p_period
402                //   || *p_period > (double)v27 && period_max < (double)*p_period )
403                //
404                // translate names:
405                // v18 = half_flake_w; v19 = half_flake_h;
406                // v23 = game_w as f32; v24 = game_h as f32;
407                // v27 = old_period; *p_period = p.period (maybe updated by adjust_flake_accel)
408                //
409                let mut triggered = false;
410
411                // 1st clause: (!old_x < new_x && ... ) i.e. cond1 && -half_flake_w * v7 > new_x
412                if cond1 && (-half_flake_w * v7) > new_x {
413                    triggered = true;
414                } else if (new_x > old_x) && (v7 * half_flake_w + v23) < new_x {
415                    // 2nd: new_x > old_x && v7*half_flake_w + screen_w < new_x
416                    triggered = true;
417                } else {
418                    // 3rd and 4th involving y:
419                    let v13_val = new_y;
420                    if (old_y > new_y && (-half_flake_h * v7) > v13_val) {
421                        triggered = true;
422                    } else if v13_val > old_y && new_y > v7 * half_flake_h + v24 {
423                        triggered = true;
424                    } else {
425                        // 5th and 6th: period checks
426                        let current_period = p.period;
427                        if old_period > current_period && period_min > current_period {
428                            triggered = true;
429                        } else if current_period > old_period && period_max < current_period {
430                            triggered = true;
431                        }
432                    }
433                }
434
435                if triggered {
436                    // set_snow_flake(this, (snow_flake *)(p_period - 1));
437                    // sub_4249B0(this, p_period - 1);
438                    Self::set_snow_flake(self, i, ori_game_w, ori_game_h);
439                    Self::adjust_after_reset(self, i, &mut rng, ori_game_w, ori_game_h);
440                }
441            } // end for each flake
442        } // end if flake_count > 0
443
444        // final sort step: if enabled still set: sort flake_ptrs by flakes[idx].period ascending
445        if self.enabled {
446            let count = self.flake_count.max(0) as usize;
447            if count > 0 {
448                // Ensure flake_ptrs[0..count] contains 0..count-1 like original initialization
449                // (Original initialization filled the pointer array with addresses of flakes; we'll store indices)
450                for i in 0..count {
451                    self.flake_ptrs[i] = i;
452                }
453                // Sort by period ascending -- equivalent to sub_430520 + sub_425FD0 comparator
454                self.flake_ptrs[..count].sort_by(|&a, &b| {
455                    let pa = self.flakes[a].period;
456                    let pb = self.flakes[b].period;
457                    pa.partial_cmp(&pb).unwrap_or(std::cmp::Ordering::Equal)
458                });
459            }
460        }
461    }

463    fn compare_flakes(a: &SnowFlake, b: &SnowFlake) -> std::cmp::Ordering {
464        a.y.partial_cmp(&b.y).unwrap_or(std::cmp::Ordering::Equal)

467    pub fn init_snow_motion(&mut self, ori_game_w: i32, ori_game_h: i32) {
468        for i in 0..1024 {
469            self.flake_ptrs[i] = i;
470        }
471
472        for i in 0..self.flake_count {
473            // a safe way
474            if let Some(_flake) = self.flakes.get_mut(i as usize) {
475                Self::set_snow_flake(self, i as usize, ori_game_w, ori_game_h);
476            }
477        }
478
479        if self.enabled && self.flake_count > 0 {
480            self.flake_ptrs[..self.flake_count as usize].sort_by(|&a, &b| {
481                let flake_a = &self.flakes[a];
482                let flake_b = &self.flakes[b];
483                Self::compare_flakes(flake_a, flake_b)
484            });
485        }
486    }

616impl SnowMotionContainer {
617    pub fn debug_dump(&self, max: usize) -> String {
618        let mut out = String::new();
619        let mut shown = 0usize;
620        for (i, m) in self.motion.iter().enumerate() {
621            if !m.enabled {
622                continue;
623            }
624            if shown >= max {
625                break;
626            }
627            out.push_str(&format!(
628                "  [snow:{}] flake_count={} color=({}, {}, {}) period=({}, {})\n",
629                i,
630                m.flake_count,
631                m.color_r,
632                m.color_g,
633                m.color_b_or_extra,
634                m.period_min,
635                m.period_max
636            ));
637            // Print a few flakes (using the current pointer order).
638            let sample = 3usize.min(m.flake_ptrs.len());
639            for k in 0..sample {
640                let idx = m.flake_ptrs[k];
641                if idx >= m.flakes.len() {
642                    continue;
643                }
644                let f = &m.flakes[idx];
645                out.push_str(&format!(
646                    "    flake[{}] var={} x={:.2} y={:.2} period={:.2}\n",
647                    idx,
648                    f.variant_idx,
649                    f.x,
650                    f.y,
651                    f.period
652                ));
653            }
654            shown += 1;
655        }
656        out
657    }
658
659    pub fn debug_enabled_count(&self) -> usize {
660        self.motion.iter().filter(|m| m.enabled).count()
661    }
