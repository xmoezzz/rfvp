12#[derive(Debug, Clone, Default, Serialize, Deserialize)]
13pub enum Nls {
14    #[default]
15    ShiftJIS = 0,
16    GBK = 1,
17    UTF8 = 2,
18}
19
20impl FromStr for Nls {
21    type Err = anyhow::Error;
22
23    fn from_str(s: &str) -> Result<Self> {
24        let lower = s.to_ascii_lowercase();
25        match lower.as_str() {
26            "sjis" => Ok(Nls::ShiftJIS),
27            "gbk" => Ok(Nls::GBK),
28            "utf8" => Ok(Nls::UTF8),
29            _ => Err(anyhow!("unknown NLS")),
30        }
31    }
32}

63    pub fn new(path: impl AsRef<Path>, nls: Nls) -> Result<Self> {
64        let mut rdr = File::open(path)?;
65        let mut buffer = Vec::new();
66        rdr.read_to_end(&mut buffer)?;
67
68        let mut parser = Parser {
69            buffer: Rc::new(buffer),

71            sys_desc_offset: 0,

73            non_volatile_global_count: 0,
74            volatile_global_count: 0,

77            game_mode_reserved: 0,
78            game_title: String::new(),

111        if offset + 3 >= self.buffer.len() {
112            return Err(anyhow!("offset out of bounds"));
113        }
114        Ok(u32::from_le_bytes([

138    /// Read a C-style string with a maximum length `len` (may contain an early NUL).
139    /// Then decode it into UTF-8 according to the configured NLS.
140    pub fn read_cstring(&self, offset: usize, len: usize) -> Result<String> {
141        if offset + len > self.buffer.len() {
142            return Err(anyhow!("offset out of bounds"));
143        }
144        let mut raw = Vec::new();
145        for i in 0..len {
146            let b = self.buffer[offset + i];
147            if b == 0 {
148                break;
149            }
150            raw.push(b);
151        }
152
153        let decoded = match self.nls {
154            Nls::ShiftJIS => {
155                let (s, _, had_err) = encoding_rs::SHIFT_JIS.decode(&raw);
156                if had_err {
157                    log::warn!("ShiftJIS decode error");
158                }
159                s
160            }
161            Nls::GBK => {
162                let (s, _, had_err) = encoding_rs::GBK.decode(&raw);
163                if had_err {
164                    log::warn!("GBK decode error");
165                }
166                s
167            }
168            Nls::UTF8 => {
169                let (s, _, had_err) = encoding_rs::UTF_8.decode(&raw);
170                if had_err {
171                    log::warn!("UTF-8 decode error");
172                }
173                s
174            }
175        };
176
177        Ok(decoded.to_string())
178    }

180    fn parse_header(&mut self) -> Result<()> {
181        let mut off: usize = 0;
182        self.sys_desc_offset = self.read_u32(off)?;
183
184        off = self.sys_desc_offset as usize;
185        self.entry_point = self.read_u32(off)?;
186        off += size_of::<u32>();
187
188        self.non_volatile_global_count = self.read_u16(off)?;
189        off += size_of::<u16>();
190
191        self.volatile_global_count = self.read_u16(off)?;
192        off += size_of::<u16>();
193
194        self.game_mode = self.read_u8(off)? as u8;
195        off += size_of::<u8>();
196
197        self.game_mode_reserved = self.read_u8(off)? as u8;
198        off += size_of::<u8>();
199
200        let title_len = self.read_u8(off)? as usize;
201        off += size_of::<u8>();
202
203        self.game_title = self.read_cstring(off, title_len)?;
204        off += title_len;
205
206        self.syscall_count = self.read_u16(off)?;
207        off += size_of::<u16>();
208
209        for i in 0..self.syscall_count {
210            let args = self.read_u8(off)?;
211            off += size_of::<u8>();
212
213            let name_len = self.read_u8(off)? as usize;
214            off += size_of::<u8>();
215
216            let name = self.read_cstring(off, name_len)?;
217            off += name_len;
218
219            self.syscalls.insert(i as usize, Syscall { args, name });
220        }
221
222        self.custom_syscall_count = self.read_u16(off)?;
223        if self.custom_syscall_count > 0 {
224            log::warn!("custom syscall count: {}", self.custom_syscall_count);
225        }
226
227        Ok(())
228    }
