5#[derive(Debug, Clone, Copy, PartialEq, Eq)]
6pub enum BlockTerm {
7    Fallthrough,
8    Jmp,
9    Jz,
10    Ret,
11    RetV,
12}

14#[derive(Debug, Clone)]
15pub struct BasicBlock {
16    pub id: usize,
17    pub start: u32,
18    pub end: u32, // exclusive (next leader or function end)
19    pub inst_indices: std::ops::Range<usize>,
20    pub preds: Vec<usize>,
21    pub succs: Vec<usize>,
22    pub term: BlockTerm,
23    pub in_depth: usize,
24    pub out_depth: usize,
25    pub is_loop_header: bool,
26}

35fn inst_stack_delta(inst: &Instruction, callee_args: &BTreeMap<u32, u8>) -> i32 {
36    match &inst.op {
37        Op::Nop | Op::InitStack { .. } | Op::Jmp { .. } | Op::Ret => 0,
38        Op::Jz { .. } => -1,
39        Op::RetV => -1,
40        Op::PushNil
41        | Op::PushTrue
42	    | Op::PushI8(_)
43	    | Op::PushI16(_)
44	    | Op::PushI32(_)
45	    | Op::PushF32(_)
46	    | Op::PushString(_)
47	    | Op::PushGlobal(_)
48	    | Op::PushStack(_) => 1,
49	    | Op::PushGlobalTable(_)
50	    | Op::PushLocalTable(_) => 0,
51        | Op::PushTop
52        | Op::PushReturn => 1,
53	    Op::PopGlobal(_) | Op::PopStack(_) => -1,
54        | Op::PopGlobalTable(_) | Op::PopLocalTable(_) => -2,
55        Op::Neg => 0,
56        Op::Add
57        | Op::Sub
58        | Op::Mul
59        | Op::Div
60        | Op::Mod
61        | Op::BitTest
62        | Op::And
63        | Op::Or
64        | Op::SetE
65        | Op::SetNE
66        | Op::SetG
67        | Op::SetLE
68        | Op::SetL
69        | Op::SetGE => -1,
70        Op::Call { target } => -(callee_args.get(target).copied().unwrap_or(0) as i32),
71        Op::Syscall { args, .. } => -(*args as i32),
72	    Op::Unknown(_) => 0,
73    }
74}

76fn block_term(last: Option<&Instruction>) -> BlockTerm {
77    match last.map(|i| &i.op) {
78        Some(Op::Jmp { .. }) => BlockTerm::Jmp,
79        Some(Op::Jz { .. }) => BlockTerm::Jz,
80        Some(Op::Ret) => BlockTerm::Ret,
81        Some(Op::RetV) => BlockTerm::RetV,
82        _ => BlockTerm::Fallthrough,
83    }
84}

86pub fn build_cfg(func: &Function, callee_args: &BTreeMap<u32, u8>) -> FunctionCfg {
87    // Leaders.
88    let mut leaders: BTreeSet<u32> = BTreeSet::new();
89    if let Some(first) = func.insts.first() {
90        leaders.insert(first.addr);
91    }
92
93    // Map addr -> inst index.
94    let mut addr_to_idx: BTreeMap<u32, usize> = BTreeMap::new();
95    for (i, inst) in func.insts.iter().enumerate() {
96        addr_to_idx.insert(inst.addr, i);
97    }
98
99    for (i, inst) in func.insts.iter().enumerate() {
100        match &inst.op {
101            Op::Jmp { target } | Op::Jz { target } => {
102                leaders.insert(*target);
103                if let Some(next) = func.insts.get(i + 1) {
104                    leaders.insert(next.addr);
105                }
106            }
107            Op::Ret | Op::RetV => {
108                if let Some(next) = func.insts.get(i + 1) {
109                    leaders.insert(next.addr);
110                }
111            }
112            _ => {}
113        }
114    }
115
116    let func_end = func
117        .insts
118        .last()
119        .map(|i| i.addr + 1)
120        .unwrap_or(func.start_addr);
121
122    // Build blocks by leader ranges.
123    let mut leader_vec: Vec<u32> = leaders.into_iter().collect();
124    leader_vec.sort();
125
126    let mut blocks: Vec<BasicBlock> = Vec::new();
127    let mut addr_to_block: BTreeMap<u32, usize> = BTreeMap::new();
128
129    for (bid, &start) in leader_vec.iter().enumerate() {
130        let end = leader_vec.get(bid + 1).copied().unwrap_or(func_end);
131        let si = addr_to_idx.get(&start).copied().unwrap_or(0);
132        let ei = addr_to_idx
133            .range(end..)
134            .next()
135            .map(|(_, &idx)| idx)
136            .unwrap_or(func.insts.len());
137
138        for (&a, _) in addr_to_idx.range(start..end) {
139            addr_to_block.insert(a, bid);
140        }
141
142        let last = if ei > 0 { func.insts.get(ei - 1) } else { None };
143        blocks.push(BasicBlock {
144            id: bid,
145            start,
146            end,
147            inst_indices: si..ei,
148            preds: Vec::new(),
149            succs: Vec::new(),
150            term: block_term(last),
151            in_depth: 0,
152            out_depth: 0,
153            is_loop_header: false,
154        });
155    }
156
157    // Fill succs.
158    for b in &mut blocks {
159        let last_idx = b.inst_indices.clone().last();
160        let last = last_idx.and_then(|i| func.insts.get(i));
161        match last.map(|i| &i.op) {
162            Some(Op::Jmp { target }) => {
163                if let Some(&tid) = addr_to_block.get(target) {
164                    b.succs.push(tid);
165                }
166            }
167            Some(Op::Jz { target }) => {
168                if let Some(&tid) = addr_to_block.get(target) {
169                    b.succs.push(tid);
170                }
171                // fallthrough
172                let ft = func
173                    .insts
174                    .get(b.inst_indices.end)
175                    .map(|i| i.addr);
176                if let Some(ft) = ft {
177                    if let Some(&fid) = addr_to_block.get(&ft) {
178                        b.succs.push(fid);
179                    }
180                }
181            }
182            Some(Op::Ret) | Some(Op::RetV) => {}
183            _ => {
184                // Fallthrough to next block by address.
185                let ft = func
186                    .insts
187                    .get(b.inst_indices.end)
188                    .map(|i| i.addr);
189                if let Some(ft) = ft {
190                    if let Some(&fid) = addr_to_block.get(&ft) {
191                        b.succs.push(fid);
192                    }
193                }
194            }
195        }
196    }
197
198    // Fill preds.
199    for b in 0..blocks.len() {
200        let succs = blocks[b].succs.clone();
201        for s in succs {
202            if let Some(sb) = blocks.get_mut(s) {
203                sb.preds.push(b);
204            }
205        }
206    }
207
208    // Compute in/out depths using a simple worklist.
209    let mut in_depth: Vec<Option<usize>> = vec![None; blocks.len()];
210    if !blocks.is_empty() {
211        in_depth[0] = Some(0);
212    }
213    let mut q: VecDeque<usize> = VecDeque::new();
214    if !blocks.is_empty() {
215        q.push_back(0);
216    }
217    let mut max_depth = 0usize;
218
219    while let Some(bid) = q.pop_front() {
220        let mut d = in_depth[bid].unwrap_or(0) as i32;
221        let b = &blocks[bid];
222
223        for i in b.inst_indices.clone() {
224            let inst = &func.insts[i];
225            d += inst_stack_delta(inst, callee_args);
226            if d < 0 {
227                d = 0;
228            }
229            max_depth = max_depth.max(d as usize);
230        }
231
232        let out = d as usize;
233        for &sid in &blocks[bid].succs {
234            match in_depth[sid] {
235                None => {
236                    in_depth[sid] = Some(out);
237                    q.push_back(sid);
238                }
239                Some(existing) => {
240                    // If mismatched, keep the max to avoid panics; we will still emit
241                    // readable code, but the stack-model is approximate.
242                    if existing != out {
243                        let newd = existing.max(out);
244                        if newd != existing {
245                            in_depth[sid] = Some(newd);
246                            q.push_back(sid);
247                        }
248                    }
249                }
250            }
251        }
252    }
253
254    for b in &mut blocks {
255        b.in_depth = in_depth[b.id].unwrap_or(0);
256        // Simulate to compute out depth.
257        let mut d = b.in_depth as i32;
258        for i in b.inst_indices.clone() {
259            let inst = &func.insts[i];
260            d += inst_stack_delta(inst, callee_args);
261            if d < 0 {
262                d = 0;
263            }
264        }
265        b.out_depth = d as usize;
266    }
267
268    // Loop header classification: any block that is target of a back-edge.
269    let mut loop_headers: BTreeSet<usize> = BTreeSet::new();
270    for b in &blocks {
271        for &s in &b.succs {
272            if let (Some(src), Some(dst)) = (blocks.get(b.id), blocks.get(s)) {
273                if dst.start < src.start {
274                    loop_headers.insert(s);
275                }
276            }
277        }
278    }
279    for b in &mut blocks {
280        b.is_loop_header = loop_headers.contains(&b.id);
281    }
282
283    FunctionCfg {
284        blocks,
285        addr_to_block,
286        max_depth,
287    }
288}
