20/// A seekable view over a contiguous byte range inside a file.
21///
22/// This is used to expose pack entries as `Read + Seek` without loading them into memory.
23#[derive(Debug)]
24pub struct SubFile {
25    file: File,
26    start: u64,
27    len: u64,
28    pos: u64,
29}
30
31impl SubFile {
32    pub fn new(mut file: File, start: u64, len: u64) -> Result<Self> {
33        file.seek(SeekFrom::Start(start))
34            .with_context(|| format!("seek pack slice start={}", start))?;
35        Ok(Self {
36            file,
37            start,
38            len,
39            pos: 0,
40        })
41    }
42
43    fn clamp_pos(&self, p: i128) -> u64 {
44        if p <= 0 {
45            return 0;
46        }
47        let p = p as u128;
48        if p >= self.len as u128 {
49            return self.len;
50        }
51        p as u64
52    }
53}
54
55impl Read for SubFile {
56    fn read(&mut self, buf: &mut [u8]) -> std::io::Result<usize> {
57        if self.pos >= self.len {
58            return Ok(0);
59        }
60        let remain = (self.len - self.pos) as usize;
61        let to_read = buf.len().min(remain);
62        // Keep the underlying FD aligned with our logical position.
63        self.file
64            .seek(SeekFrom::Start(self.start + self.pos))?;
65        let n = self.file.read(&mut buf[..to_read])?;
66        self.pos += n as u64;
67        Ok(n)
68    }
69}
70
71impl Seek for SubFile {
72    fn seek(&mut self, pos: SeekFrom) -> std::io::Result<u64> {
73        let next = match pos {
74            SeekFrom::Start(off) => self.clamp_pos(off as i128),
75            SeekFrom::End(delta) => self.clamp_pos(self.len as i128 + delta as i128),
76            SeekFrom::Current(delta) => self.clamp_pos(self.pos as i128 + delta as i128),
77        };
78        self.pos = next;
79        self.file
80            .seek(SeekFrom::Start(self.start + self.pos))?;
81        Ok(self.pos)
82    }
83}

92pub struct VfsFile {

94    pub folder_name: String,
95    pub file_count: u64,

97    pub entries: HashMap<String, VfsEntry>,

101impl VfsFile {
102    pub fn new(path: PathBuf, folder_name: String, nls: Nls) -> anyhow::Result<Self> {
103        let (file_count, filename_table_size, entries) = VfsFile::parse(&path, nls)
104            .with_context(|| format!("parse pack {}", path.display()))?;
105
106        Ok(VfsFile {
107            path,
108            folder_name,
109            file_count,
110            filename_table_size,
111            entries,
112            nls: nls.clone(),
113        })
114    }
115
116    fn read_u32le(reader: &mut (impl Read + Seek), offset: u64) -> Result<u32> {
117        let mut buffer = [0u8; 4];
118        reader.seek(SeekFrom::Start(offset))?;
119        reader.read_exact(&mut buffer)?;
120        Ok(u32::from_le_bytes(buffer))
121    }
122
123    /// Read C-style NUL-terminated strings from `[offset, offset+size)`.
124    ///
125    /// Returns a map: `string_start_offset_in_table -> decoded string`.
126    fn read_filename_table(
127        reader: &mut (impl Read + Seek),
128        offset: u64,
129        size: u64,
130        nls: Nls,
131    ) -> Result<HashMap<u64, String>> {
132        let mut buffer = vec![0u8; size as usize];
133        reader.seek(SeekFrom::Start(offset))?;
134        reader.read_exact(&mut buffer)?;
135
136        let mut results = HashMap::new();
137        let mut start = 0usize;
138        for (i, &b) in buffer.iter().enumerate() {
139            if b == 0 {
140                let s = match nls {
141                    Nls::ShiftJIS => {
142                        let (s, _, _) = encoding_rs::SHIFT_JIS.decode(&buffer[start..i]);
143                        s.to_string()
144                    }
145                    Nls::GBK => {
146                        let (s, _, _) = encoding_rs::GBK.decode(&buffer[start..i]);
147                        s.to_string()
148                    }
149                    Nls::UTF8 => String::from_utf8_lossy(&buffer[start..i]).into_owned(),
150                };
151                results.insert(start as u64, s);
152                start = i + 1;
153            }
154        }
155        Ok(results)
156    }
157
158    /// Parse `.bin` package:
159    /// - u32 file_count
160    /// - u32 filename_table_size
161    /// - file_count entries, each 12 bytes: {u32 name_off, u32 data_off, u32 data_size}
162    /// - filename table (NUL-terminated strings)
163    pub fn parse(
164        path: impl AsRef<Path>,
165        nls: Nls,
166    ) -> Result<(u64, u64, HashMap<String, VfsEntry>)> {
167        let path = path.as_ref();
168        if !path.exists() {
169            bail!("pack does not exist: {}", path.display());
170        }
171
172        let mut file = File::open(path).with_context(|| format!("open {}", path.display()))?;
173
174        let mut offset = 0u64;
175        let file_count = Self::read_u32le(&mut file, offset)? as u64;
176        offset += size_of::<u32>() as u64;
177
178        let filename_table_size = Self::read_u32le(&mut file, offset)? as u64;
179        offset += size_of::<u32>() as u64;
180
181        // entries begin at offset=8
182        let entries_offset = offset;
183        let filename_table_offset = entries_offset + file_count * 12;
184
185        let filename_table =
186            Self::read_filename_table(&mut file, filename_table_offset, filename_table_size, nls)?;
187
188        // entry table
189        file.seek(SeekFrom::Start(entries_offset))?;
190        let mut entries = HashMap::new();
191        let mut cur = entries_offset;
192        for _ in 0..file_count {
193            let name_off = Self::read_u32le(&mut file, cur)? as u64;
194            cur += 4;
195            let data_off = Self::read_u32le(&mut file, cur)? as u64;
196            cur += 4;
197            let data_size = Self::read_u32le(&mut file, cur)? as u64;
198            cur += 4;
199
200            if let Some(name) = filename_table.get(&name_off) {
201                entries.insert(
202                    name.clone(),
203                    VfsEntry {
204                        offset: data_off,
205                        size: data_size,
206                    },
207                );
208            }
209        }
210
211        Ok((file_count, filename_table_size, entries))
212    }

214    /// Open an entry as a seekable stream.
215    ///
216    /// Resolution order:
217    /// 1) Prefer loose file override at `<game_root>/<folder_name>/<name>` if it exists.
218    /// 2) Fallback to reading from `<game_root>/<folder_name>.bin` pack.
219    pub fn open_stream(&self, name: &str) -> Result<VfsStream> {
220        let override_path = app_base_path().join(&self.folder_name).join(name);
221        if override_path.get_path().exists() {
222            let f = File::open(override_path.get_path())
223                .with_context(|| format!("open override file {}", override_path.get_path().display()))?;
224            return Ok(Box::new(f));
225        }
226
227        let ent = self
228            .entries
229            .get(name)
230            .ok_or_else(|| anyhow::anyhow!("file not found in pack {}: {}", self.path.display(), name))?;
231        let f = File::open(&self.path)
232            .with_context(|| format!("open pack file {}", self.path.display()))?;
233        let sub = SubFile::new(f, ent.offset, ent.size)
234            .with_context(|| format!("create SubFile slice for {}", name))?;
235        Ok(Box::new(sub))
236    }
237
238    /// Legacy convenience: read an entry fully into memory.
239    pub fn read_file(&self, name: &str) -> Result<Vec<u8>> {
240        let mut r = self.open_stream(name)?;
241        let mut buf = Vec::with_capacity(self.entries.get(name).map(|e| e.size as usize).unwrap_or(0));
242        r.read_to_end(&mut buf)
243            .with_context(|| format!("read all bytes for {}", name))?;
244        Ok(buf)
245    }
246
247    pub fn save(&self, name: &str, content: Vec<u8>) -> Result<()> {
248        let mut file = File::create(app_base_path().join(&self.folder_name).join(name).get_path())
249            .with_context(|| format!("create override file for {}", name))?;
250        file.write_all(&content)
251            .with_context(|| format!("write override file for {}", name))?;
252        Ok(())
253    }
254}

256#[derive(Debug)]
257pub struct Vfs {
258    pub files: HashMap<String, VfsFile>,
259    pub nls: Nls,
260}
261
262impl Default for Vfs {
263    fn default() -> Self {
264        Vfs::new(Nls::ShiftJIS).expect("default Vfs initialization")
265    }
266}
267
268impl Vfs {
269    pub fn new(nls: Nls) -> Result<Vfs> {
270        let path = app_base_path().join("*.bin");
271        let mut files = HashMap::new();
272        for entry in glob(path.get_path().to_str().unwrap())? {
273            if let Ok(path) = entry {
274                let filename = path.file_stem().unwrap().to_string_lossy();
275                if filename.is_empty() {
276                    continue;
277                }
278                let folder_name = filename.to_string();
279                if let Ok(vf) = VfsFile::new(path, folder_name.to_ascii_lowercase().clone(), nls) {
280                    files.insert(folder_name.to_ascii_lowercase(), vf);
281                }
282            }
283        }
284
285        Ok(Vfs { files, nls: nls.clone() })
286    }
287
288    /// Open a path as a seekable stream.
289    ///
290    /// Resolution order (as requested):
291    /// 1) Try direct filesystem open at `<game_root>/<path>`.
292    /// 2) If it fails and `path` is `folder/name...`, look for `<folder>.bin` and open the entry.
293    pub fn open_stream(&self, path: &str) -> Result<VfsStream> {
294        // 1) direct file override
295        let fs_path = app_base_path().join(path);
296        if fs_path.get_path().exists() {
297            let f = File::open(fs_path.get_path())
298                .with_context(|| format!("open file {}", fs_path.get_path().display()))?;
299            return Ok(Box::new(f));
300        }
301
302        // 2) packed: folder/name...
303        let (folder, inner) = path
304            .split_once('/')
305            .ok_or_else(|| anyhow::anyhow!("file not found: {}", path))?;
306
307        let vf = self
308            .files
309            .get(&folder.to_ascii_lowercase())
310            .ok_or_else(|| anyhow::anyhow!("pack not found for folder '{}' (missing {}.bin)", folder, folder))?;
311        vf.open_stream(inner)
312    }
313
314    /// Legacy convenience: read a path fully into memory.
315    ///
316    /// This retains the old signature but now follows the same resolution order as `open_stream`.
317    pub fn read_file(&self, path: &str) -> Result<Vec<u8>> {
318        let mut r = self.open_stream(path)?;
319        let mut buf = Vec::new();
320        r.read_to_end(&mut buf)
321            .with_context(|| format!("read all bytes for {}", path))?;
322        Ok(buf)
323    }
324
325    pub fn save(&self, path: &str, content: Vec<u8>) -> Result<()> {
326        let (folder, name) = path
327            .split_once('/')
328            .ok_or_else(|| anyhow::anyhow!("invalid vfs path (expected folder/name): {}", path))?;
329        let file = self
330            .files
331            .get(folder)
332            .ok_or_else(|| anyhow::anyhow!("missing vfs pack for folder: {}", folder))?;
333        file.save(name, content)
334    }
335
336    /// Find loose `.ani` files in `<game_root>/graph`.
337    ///
338    /// Note: this does not (yet) scan packs. It's primarily used for cursor assets.
339    pub fn find_ani(&self) -> Result<Vec<PathBuf>> {
340        let path = app_base_path().join("graph").join("*.ani");
341
342        let matches: Vec<_> = glob(path.get_path().to_str().unwrap())?.flatten().collect();
343        Ok(matches)
344    }
345}
