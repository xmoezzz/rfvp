1use std::collections::VecDeque;
2
3pub enum ThreadRequest {
4    /// start a new thread with the given id and address
5    Start(u32, u32),
6    /// wait for the given time
7    Wait(u32),
8    /// wait until dissolve is completed / static
9    DissolveWait(),
10    /// sleep for the given time, depercated request
11    Sleep(u32),
12    /// raise the threads which are waiting for the given time, depercated request
13    Raise(u32),
14    /// yield the current thread
15    Next(),
16    /// exit the corresponding thread, None is for all threads,
17    /// If all threads are exited, the game will be impossible to manipulate through the script engine
18    Exit(Option<u32>),
19    ShouldBreak(),
20}
21
22#[derive(Default)]
23pub struct ThreadWrapper {
24    requests: VecDeque<ThreadRequest>,
25}
26
27
28impl ThreadWrapper {
29    pub fn new() -> Self {
30        Default::default()
31    }
32
33    pub fn pop(&mut self) -> Option<ThreadRequest> {
34        self.requests.pop_front()
35    }
36
37    pub fn thread_start(&mut self, id: u32, addr: u32) {
38        self.requests.push_back(ThreadRequest::Start(id, addr));
39    }
40
41    pub fn thread_wait(&mut self, time: u32) {
42        self.requests.push_back(ThreadRequest::Wait(time));
43    }
44
45    pub fn dissolve_wait(&mut self) {
46        self.requests.push_back(ThreadRequest::DissolveWait());
47    }
48
49    pub fn thread_sleep(&mut self, time: u32) {
50        self.requests.push_back(ThreadRequest::Sleep(time));
51    }
52
53    pub fn thread_raise(&mut self, time: u32) {
54        self.requests.push_back(ThreadRequest::Raise(time));
55    }
56
57    pub fn thread_next(&mut self) {
58        self.requests.push_back(ThreadRequest::Next());
59    }
60
61    pub fn thread_exit(&mut self, id: Option<u32>) {
62        self.requests.push_back(ThreadRequest::Exit(id));
63    }
64
65    pub fn should_break(&mut self) {
66        self.requests.push_back(ThreadRequest::ShouldBreak());
67    }
68}
