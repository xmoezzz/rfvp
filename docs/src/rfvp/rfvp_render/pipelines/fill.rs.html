5pub struct FillPipeline {
6    pipeline: wgpu::RenderPipeline,
7}
8
9impl FillPipeline {
10    pub fn new(device: &wgpu::Device, target_format: wgpu::TextureFormat) -> Self {
11        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
12            label: Some("rfvp_render.fill_shader"),
13            source: wgpu::ShaderSource::Wgsl(include_str!("fill.wgsl").into()),
14        });
15
16        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
17            label: Some("rfvp_render.fill_pipeline_layout"),
18            bind_group_layouts: &[],
19            push_constant_ranges: &[wgpu::PushConstantRange {
20                stages: wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT,
21                range: 0..80,
22            }],
23        });
24
25        let vertex_layout = wgpu::VertexBufferLayout {
26            array_stride: std::mem::size_of::<PosVertex>() as u64,
27            step_mode: wgpu::VertexStepMode::Vertex,
28            attributes: &[wgpu::VertexAttribute {
29                offset: 0,
30                shader_location: 0,
31                format: wgpu::VertexFormat::Float32x3,
32            }],
33        };
34
35        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
36            label: Some("rfvp_render.fill_pipeline"),
37            layout: Some(&pipeline_layout),
38            vertex: wgpu::VertexState {
39                module: &shader,
40                entry_point: "vs_main",
41                buffers: &[vertex_layout],
42            },
43            fragment: Some(wgpu::FragmentState {
44                module: &shader,
45                entry_point: "fs_main",
46                targets: &[Some(wgpu::ColorTargetState {
47                    format: target_format,
48                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
49                    write_mask: wgpu::ColorWrites::ALL,
50                })],
51            }),
52            primitive: wgpu::PrimitiveState {
53                topology: wgpu::PrimitiveTopology::TriangleList,
54                strip_index_format: None,
55                front_face: wgpu::FrontFace::Ccw,
56                cull_mode: None,
57                polygon_mode: wgpu::PolygonMode::Fill,
58                unclipped_depth: false,
59                conservative: false,
60            },
61            depth_stencil: None,
62            multisample: wgpu::MultisampleState::default(),
63            multiview: None,
64        });
65
66        Self { pipeline }
67    }
68
69    pub fn draw<'a>(
70        &'a self,
71        pass: &mut wgpu::RenderPass<'a>,
72        src: VertexSource<'a>,
73        transform: Mat4,
74        color: Vec4,
75    ) {
76        pass.set_pipeline(&self.pipeline);
77
78        let mut pc = [0u8; 80];
79        let m = transform.to_cols_array();
80        pc[0..64].copy_from_slice(bytemuck::bytes_of(&m));
81        let c = color.to_array();
82        pc[64..80].copy_from_slice(bytemuck::bytes_of(&c));
83        pass.set_push_constants(wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT, 0, &pc);
84
85        match src {
86            VertexSource::VertexBuffer { vertex_buffer, vertices, instances } => {
87                pass.set_vertex_buffer(0, vertex_buffer.slice(..));
88                pass.draw(vertices, instances);
89            }
90            VertexSource::VertexIndexBuffer { vertex_buffer, index_buffer, indices, instances } => {
91                pass.set_vertex_buffer(0, vertex_buffer.slice(..));
92                pass.set_index_buffer(index_buffer.slice(..), wgpu::IndexFormat::Uint16);
93                pass.draw_indexed(indices, 0, instances);
94            }
95        }
96    }
97}
