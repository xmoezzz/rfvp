8pub struct VertexBuffer<T> {
9    buffer: wgpu::Buffer,
10    capacity: u32,
11    _marker: PhantomData<T>,
12}
13
14impl<T: Pod> VertexBuffer<T> {
15    pub fn new_updatable(resources: &GpuCommonResources, capacity: u32, label: Option<&str>) -> Self {
16        let size_bytes = (capacity as usize).saturating_mul(std::mem::size_of::<T>()) as u64;
17        let buffer = resources.device.create_buffer(&wgpu::BufferDescriptor {
18            label,
19            size: size_bytes.max(4),
20            usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
21            mapped_at_creation: false,
22        });
23        Self { buffer, capacity, _marker: PhantomData }
24    }
25
26    pub fn capacity(&self) -> u32 {
27        self.capacity
28    }
29
30    pub fn buffer(&self) -> &wgpu::Buffer {
31        &self.buffer
32    }
33
34    pub fn write(&self, queue: &wgpu::Queue, data: &[T]) {
35        queue.write_buffer(&self.buffer, 0, bytemuck::cast_slice(data));
36    }
37
38    pub fn vertex_source_slice<'a>(&'a self, vertices: Range<u32>) -> VertexSource<'a> {
39        VertexSource::VertexBuffer {
40            vertex_buffer: &self.buffer,
41            vertices,
42            instances: 0..1,
43        }
44    }
45}
