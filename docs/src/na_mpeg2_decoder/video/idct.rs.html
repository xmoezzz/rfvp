3// Constants for BIT_DEPTH=8 path.
4const W1: i64 = 22725;
5const W2: i64 = 21407;
6const W3: i64 = 19266;
7const W4: i64 = 16383;
8const W5: i64 = 12873;
9const W6: i64 = 8867;
10const W7: i64 = 4520;
11
12const ROW_SHIFT: i64 = 11;
13const COL_SHIFT: i64 = 20;
14const DC_SHIFT: i64 = 3;

16#[inline]
17fn idct_row_cond_dc(row: &mut [i16; 8]) {
18    // This is the generic path (no DC-only shortcut).
19    let r0 = row[0] as i64;
20    let r1 = row[1] as i64;
21    let r2 = row[2] as i64;
22    let r3 = row[3] as i64;
23    let r4 = row[4] as i64;
24    let r5 = row[5] as i64;
25    let r6 = row[6] as i64;
26    let r7 = row[7] as i64;
27
28    let mut a0 = W4 * r0 + (1 << (ROW_SHIFT + 0 - 1));
29    let mut a1 = a0;
30    let mut a2 = a0;
31    let mut a3 = a0;
32
33    a0 += W2 * r2;
34    a1 += W6 * r2;
35    a2 -= W6 * r2;
36    a3 -= W2 * r2;
37
38    let mut b0 = W1 * r1 + W3 * r3;
39    let mut b1 = W3 * r1 - W7 * r3;
40    let mut b2 = W5 * r1 - W1 * r3;
41    let mut b3 = W7 * r1 - W5 * r3;
42
43    if (r4 | r5 | r6 | r7) != 0 {
44        a0 += W4 * r4 + W6 * r6;
45        a1 += -W4 * r4 - W2 * r6;
46        a2 += -W4 * r4 + W2 * r6;
47        a3 += W4 * r4 - W6 * r6;
48
49        b0 += W5 * r5 + W7 * r7;
50        b1 += -W1 * r5 - W5 * r7;
51        b2 += W7 * r5 + W3 * r7;
52        b3 += W3 * r5 - W1 * r7;
53    }
54
55    row[0] = ((a0 + b0) >> ROW_SHIFT) as i16;
56    row[7] = ((a0 - b0) >> ROW_SHIFT) as i16;
57    row[1] = ((a1 + b1) >> ROW_SHIFT) as i16;
58    row[6] = ((a1 - b1) >> ROW_SHIFT) as i16;
59    row[2] = ((a2 + b2) >> ROW_SHIFT) as i16;
60    row[5] = ((a2 - b2) >> ROW_SHIFT) as i16;
61    row[3] = ((a3 + b3) >> ROW_SHIFT) as i16;
62    row[4] = ((a3 - b3) >> ROW_SHIFT) as i16;
63}

65#[inline]
66fn idct_cols(col: [i16; 8]) -> [i64; 8] {
67    let c0 = col[0] as i64;
68    let c1 = col[1] as i64;
69    let c2 = col[2] as i64;
70    let c3 = col[3] as i64;
71    let c4 = col[4] as i64;
72    let c5 = col[5] as i64;
73    let c6 = col[6] as i64;
74    let c7 = col[7] as i64;
75
76    // a0 = W4 * (col0 + ((1<<(COL_SHIFT-1))/W4));
77    let add = (1i64 << (COL_SHIFT - 1)) / W4;
78    let mut a0 = W4 * (c0 + add);
79    let mut a1 = a0;
80    let mut a2 = a0;
81    let mut a3 = a0;
82
83    a0 += W2 * c2;
84    a1 += W6 * c2;
85    a2 -= W6 * c2;
86    a3 -= W2 * c2;
87
88    let mut b0 = W1 * c1 + W3 * c3;
89    let mut b1 = W3 * c1 - W7 * c3;
90    let mut b2 = W5 * c1 - W1 * c3;
91    let mut b3 = W7 * c1 - W5 * c3;
92
93    if (c4 | c5 | c6 | c7) != 0 {
94        a0 += W4 * c4 + W6 * c6;
95        a1 += -W4 * c4 - W2 * c6;
96        a2 += -W4 * c4 + W2 * c6;
97        a3 += W4 * c4 - W6 * c6;
98
99        b0 += W5 * c5 + W7 * c7;
100        b1 += -W1 * c5 - W5 * c7;
101        b2 += W7 * c5 + W3 * c7;
102        b3 += W3 * c5 - W1 * c7;
103    }
104
105    [a0 + b0, a1 + b1, a2 + b2, a3 + b3, a3 - b3, a2 - b2, a1 - b1, a0 - b0]
106}

108pub fn simple_idct_put(dest: &mut [u8], stride: usize, block: &mut [i16; 64]) {
109    // Row transforms (in-place)
110    for y in 0..8 {
111        let mut row = [0i16; 8];
112        for x in 0..8 {
113            row[x] = block[y * 8 + x];
114        }
115        idct_row_cond_dc(&mut row);
116        for x in 0..8 {
117            block[y * 8 + x] = row[x];
118        }
119    }
120
121    // Column transforms + store
122    #[allow(clippy::erasing_op)]
123    for x in 0..8 {
124        let col = [
125            block[0 * 8 + x],
126            block[1 * 8 + x],
127            block[2 * 8 + x],
128            block[3 * 8 + x],
129            block[4 * 8 + x],
130            block[5 * 8 + x],
131            block[6 * 8 + x],
132            block[7 * 8 + x],
133        ];
134        let out = idct_cols(col);
135        for y in 0..8 {
136            let v = (out[y] >> COL_SHIFT) as i32;
137            dest[y * stride + x] = clip_u8(v);
138        }
139    }
140}

142pub fn simple_idct_add(dest: &mut [u8], stride: usize, block: &mut [i16; 64]) {
143    for y in 0..8 {
144        let mut row = [0i16; 8];
145        for x in 0..8 {
146            row[x] = block[y * 8 + x];
147        }
148        idct_row_cond_dc(&mut row);
149        for x in 0..8 {
150            block[y * 8 + x] = row[x];
151        }
152    }
153
154    for x in 0..8 {
155        let col = [
156            block[0 * 8 + x],
157            block[1 * 8 + x],
158            block[2 * 8 + x],
159            block[3 * 8 + x],
160            block[4 * 8 + x],
161            block[5 * 8 + x],
162            block[6 * 8 + x],
163            block[7 * 8 + x],
164        ];
165        let out = idct_cols(col);
166        for y in 0..8 {
167            let add = (out[y] >> COL_SHIFT) as i32;
168            let cur = dest[y * stride + x] as i32;
169            dest[y * stride + x] = clip_u8(cur + add);
170        }
171    }
172}
