5#[derive(Clone, Copy, Debug, PartialEq, Eq)]
6pub enum MotionOp {
7    Put,
8    Avg,
9}
10
11#[inline]
12fn op_store(op: MotionOp, dst: &mut u8, val: u8) {
13    match op {
14        MotionOp::Put => *dst = val,
15        MotionOp::Avg => *dst = avg_u8(*dst, val),
16    }
17}
18
19#[inline]
20fn interp_2tap(a: u8, b: u8) -> u8 {
21    avg_u8(a, b)
22}
23
24#[inline]
25fn interp_4tap(a: u8, b: u8, c: u8, d: u8) -> u8 {
26    // Bilinear interpolation at half-pel in both directions.
27    // Exact integer rounding: (a + b + c + d + 2) >> 2.
28    ((a as u16 + b as u16 + c as u16 + d as u16 + 2) >> 2) as u8
29}

32pub struct MotionCompensator {

37impl MotionCompensator {
38    pub fn new() -> Self {
39        Self {
40            edge_y: vec![0u8; 17 * 17],
41            edge_uv: vec![0u8; 17 * 17],
42        }
43    }

45    fn mc_plane(
46        dst: &mut [u8],
47        dst_stride: usize,
48        src: &[u8],
49        src_stride: usize,
50        w: isize,
51        h: isize,
52        dst_x: usize,
53        dst_y: usize,
54        src_x: isize,
55        src_y: isize,
56        bw: usize,
57        bh: usize,
58        dxy: u32,
59        op: MotionOp,
60        scratch: &mut [u8],
61        scratch_stride: usize,
62    ) {
63        emulated_edge_mc(
64            scratch,
65            src,
66            scratch_stride as isize,
67            src_stride as isize,
68            bw + 1,
69            bh + 1,
70            src_x,
71            src_y,
72            w,
73            h,
74        );
75
76        let dx = (dxy & 1) as usize;
77        let dy = ((dxy >> 1) & 1) as usize;
78
79        for yy in 0..bh {
80            for xx in 0..bw {
81                let base = yy * scratch_stride + xx;
82                let a = scratch[base];
83                let val = match (dx, dy) {
84                    (0, 0) => a,
85                    (1, 0) => interp_2tap(a, scratch[base + 1]),
86                    (0, 1) => interp_2tap(a, scratch[base + scratch_stride]),
87                    (1, 1) => interp_4tap(
88                        a,
89                        scratch[base + 1],
90                        scratch[base + scratch_stride],
91                        scratch[base + scratch_stride + 1],
92                    ),
93                    _ => a,
94                };
95                let di = (dst_y + yy) * dst_stride + (dst_x + xx);
96                op_store(op, &mut dst[di], val);
97            }
98        }
99    }

101    #[allow(clippy::too_many_arguments)]
102    fn mpeg_motion_internal(
103        &mut self,
104        cur: &mut Frame,
105        op: MotionOp,
106        ref_frame: &Frame,
107        field_based: bool,
108        bottom_field: bool,
109        field_select: bool,
110        mb_x: usize,
111        mb_y: usize,
112        motion_x: i32,
113        motion_y: i32,
114        h: usize,
115        is_16x8: bool,
116    ) {
117        let (chroma_x_shift, chroma_y_shift) = cur.chroma_shifts();
118
119        let field_based_i = if field_based { 1 } else { 0 };
120        let block_y_half = (field_based_i | if is_16x8 { 1 } else { 0 }) as usize;
121
122        let dxy = (((motion_y & 1) << 1) | (motion_x & 1)) as u32;
123        let src_x = (mb_x as i32) * 16 + (motion_x >> 1);
124        let src_y = ((mb_y as i32) << (4 - block_y_half)) + (motion_y >> 1);
125
126        let (uvdxy, uvsrc_x, uvsrc_y) = if chroma_y_shift != 0 {
127            let mx = motion_x / 2;
128            let my = motion_y / 2;
129            (
130                (((my & 1) << 1) | (mx & 1)) as u32,
131                (mb_x as i32) * 8 + (mx >> 1),
132                ((mb_y as i32) << (3 - block_y_half)) + (my >> 1),
133            )
134        } else {
135            if chroma_x_shift != 0 {
136                // 4:2:2
137                let mx = motion_x / 2;
138                (
139                    (((motion_y & 1) << 1) | (mx & 1)) as u32,
140                    (mb_x as i32) * 8 + (mx >> 1),
141                    src_y,
142                )
143            } else {
144                // 4:4:4
145                (dxy, src_x, src_y)
146            }
147        };
148
149        // Destination view
150        let (base_y, stride_y) = if field_based {
151            let base = if bottom_field { cur.linesize_y } else { 0 };
152            (base, cur.linesize_y * 2)
153        } else {
154            (0, cur.linesize_y)
155        };
156        let (base_uv, stride_uv) = if field_based {
157            let base = if bottom_field { cur.linesize_u } else { 0 };
158            (base, cur.linesize_u * 2)
159        } else {
160            (0, cur.linesize_u)
161        };
162
163        // Reference view.
164        // In field-based motion compensation, the reference is accessed with doubled stride
165        // (sampling every other line). Field selection only adds a one-line offset.
166        let ref_stride_y = if field_based {
167            ref_frame.linesize_y * 2
168        } else {
169            ref_frame.linesize_y
170        };
171        let ref_stride_uv = if field_based {
172            ref_frame.linesize_u * 2
173        } else {
174            ref_frame.linesize_u
175        };
176        let ref_base_y = if field_select { ref_frame.linesize_y } else { 0 };
177        let ref_base_uv = if field_select { ref_frame.linesize_u } else { 0 };
178
179        // Destination coordinates.
180        let dst_x = mb_x * 16;
181        let dst_y = if field_based { mb_y * 8 } else { mb_y * 16 };
182
183        // Luma
184        {
185            let dst_plane = &mut cur.data_y;
186            let src_plane = &ref_frame.data_y;
187            Self::mc_plane(
188                &mut dst_plane[base_y..],
189                stride_y,
190                &src_plane[ref_base_y..],
191                ref_stride_y,
192                ref_frame.width as isize,
193                ref_frame.height as isize,
194                dst_x,
195                dst_y,
196                src_x as isize,
197                src_y as isize,
198                16,
199                h,
200                dxy,
201                op,
202                &mut self.edge_y,
203                17,
204            );
205        }
206
207        let bw_uv = 16 >> chroma_x_shift;
208        let bh_uv = h >> chroma_y_shift;
209        let dst_x_uv = mb_x * bw_uv;
210        let dst_y_uv = if field_based { mb_y * (8 >> chroma_y_shift) } else { mb_y * (16 >> chroma_y_shift) };
211
212        // U
213        {
214            let dst_u = &mut cur.data_u;
215            let src_u = &ref_frame.data_u;
216            Self::mc_plane(
217                &mut dst_u[base_uv..],
218                stride_uv,
219                &src_u[ref_base_uv..],
220                ref_stride_uv,
221                (ref_frame.width >> chroma_x_shift) as isize,
222                (ref_frame.height >> chroma_y_shift) as isize,
223                dst_x_uv,
224                dst_y_uv,
225                uvsrc_x as isize,
226                uvsrc_y as isize,
227                bw_uv,
228                bh_uv,
229                uvdxy,
230                op,
231                &mut self.edge_uv,
232                17,
233            );
234        }
235
236        // V
237        {
238            let dst_v = &mut cur.data_v;
239            let src_v = &ref_frame.data_v;
240            Self::mc_plane(
241                &mut dst_v[base_uv..],
242                stride_uv,
243                &src_v[ref_base_uv..],
244                ref_stride_uv,
245                (ref_frame.width >> chroma_x_shift) as isize,
246                (ref_frame.height >> chroma_y_shift) as isize,
247                dst_x_uv,
248                dst_y_uv,
249                uvsrc_x as isize,
250                uvsrc_y as isize,
251                bw_uv,
252                bh_uv,
253                uvdxy,
254                op,
255                &mut self.edge_uv,
256                17,
257            );
258        }
259    }

261    /// Motion compensation driver corresponding to MPEG-1/2 subset of `mpv_motion_internal()`.
262    #[allow(clippy::too_many_arguments)]
263    pub fn mpv_motion(
264        &mut self,
265        cur: &mut Frame,
266        ref_frame: &Frame,
267        dir: usize,
268        op: MotionOp,
269        mv_type: i32,
270        picture_structure: i32,
271        mb_x: usize,
272        mb_y: usize,
273        mv: &[[[i32; 2]; 4]; 2],
274        field_select: &[[i32; 2]; 2],
275        first_field: bool,
276    ) {
277        match mv_type {
278            0 => {
279                // MV_TYPE_16X16
280                self.mpeg_motion_internal(
281                    cur,
282                    op,
283                    ref_frame,
284                    false,
285                    false,
286                    false,
287                    mb_x,
288                    mb_y,
289                    mv[dir][0][0],
290                    mv[dir][0][1],
291                    16,
292                    false,
293                );
294            }
295            2 => {
296                // MV_TYPE_FIELD
297                if picture_structure == 3 {
298                    // Frame picture: process top and bottom fields separately.
299                    for i in 0..2 {
300                        self.mpeg_motion_internal(
301                            cur,
302                            op,
303                            ref_frame,
304                            true,
305                            i == 1,
306                            field_select[dir][i] != 0,
307                            mb_x,
308                            mb_y,
309                            mv[dir][i][0],
310                            mv[dir][i][1],
311                            8,
312                            false,
313                        );
314                    }
315                } else {
316                    // Field picture: single motion.
317                    self.mpeg_motion_internal(
318                        cur,
319                        op,
320                        ref_frame,
321                        false,
322                        false,
323                        field_select[dir][0] != 0,
324                        mb_x,
325                        mb_y >> 1,
326                        mv[dir][0][0],
327                        mv[dir][0][1],
328                        16,
329                        false,
330                    );
331                }
332            }
333            1 => {
334                // MV_TYPE_16X8
335                for i in 0..2 {
336                    self.mpeg_motion_internal(
337                        cur,
338                        op,
339                        ref_frame,
340                        false,
341                        false,
342                        field_select[dir][i] != 0,
343                        mb_x,
344                        (mb_y & !1) + i,
345                        mv[dir][i][0],
346                        mv[dir][i][1],
347                        8,
348                        true,
349                    );
350                }
351            }
352            3 => {
353                // MV_TYPE_DMV
354                if picture_structure == 3 {
355                    // Frame: two passes.
356                    let mut op2 = op;
357                    for i in 0..2 {
358                        for j in 0..2 {
359                            self.mpeg_motion_internal(
360                                cur,
361                                op2,
362                                ref_frame,
363                                true,
364                                j == 1,
365                                (j ^ i) != 0,
366                                mb_x,
367                                mb_y,
368                                mv[dir][2 * i + j][0],
369                                mv[dir][2 * i + j][1],
370                                8,
371                                false,
372                            );
373                        }
374                        op2 = MotionOp::Avg;
375                    }
376                } else {
377                    // Field picture.
378                    let mut op2 = op;
379                    for i in 0..2 {
380                        self.mpeg_motion_internal(
381                            cur,
382                            op2,
383                            ref_frame,
384                            false,
385                            false,
386                            picture_structure != (i as i32 + 1),
387                            mb_x,
388                            mb_y >> 1,
389                            mv[dir][2 * i][0],
390                            mv[dir][2 * i][1],
391                            16,
392                            false,
393                        );
394                        op2 = MotionOp::Avg;
395                        let _ = first_field;
396                    }
397                }
398            }
399            _ => {}
400        }
401    }
