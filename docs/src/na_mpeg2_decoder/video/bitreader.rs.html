3/// MSB-first bitreader equivalent to `GetBitContext`.
4#[derive(Clone, Debug)]
5pub struct GetBits<'a> {
6    buf: &'a [u8],
7    size_in_bits: usize,
8    bit_pos: usize,
9}
10
11impl<'a> GetBits<'a> {
12    pub fn init(buf: &'a [u8]) -> Self {
13        Self { buf, size_in_bits: buf.len() * 8, bit_pos: 0 }
14    }
15
16    #[inline]
17    pub fn bits_left(&self) -> i32 {
18        self.size_in_bits as i32 - self.bit_pos as i32
19    }
20
21    #[inline]
22    pub fn bits_count(&self) -> usize {
23        self.bit_pos
24    }
25
26    #[inline]
27    pub fn align(&mut self) {
28        self.bit_pos = (self.bit_pos + 7) & !7;
29    }
30
31    #[inline]
32    pub fn skip_bits(&mut self, n: usize) {
33        self.bit_pos = self.bit_pos.saturating_add(n);
34    }
35
36    #[inline]
37    pub fn skip_bits1(&mut self) {
38        self.skip_bits(1);
39    }
40
41    #[inline]
42    pub fn get_bits1(&mut self) -> u32 {
43        self.get_bits(1)
44    }
45
46    #[inline]
47    pub fn show_bits(&self, n: usize) -> u32 {
48        debug_assert!(n <= 32);
49        if n == 0 {
50            return 0;
51        }
52        let byte_pos = self.bit_pos >> 3;
53        let bit_off = self.bit_pos & 7;
54
55        // Load up to 8 bytes.
56        let mut acc: u64 = 0;
57        for i in 0..8 {
58            let b = self.buf.get(byte_pos + i).copied().unwrap_or(0);
59            acc = (acc << 8) | (b as u64);
60        }
61        // Align so that the current bit is at the top.
62        let shift = 64 - 8 * 8 + bit_off; // = bit_off
63        let acc = acc << shift;
64        let val = (acc >> (64 - n)) as u32;
65        val
66    }
67
68    #[inline]
69    pub fn show_bits_long(&self, n: usize) -> u32 {
70        self.show_bits(n)
71    }
72
73    #[inline]
74    pub fn get_bits(&mut self, n: usize) -> u32 {
75        let v = self.show_bits(n);
76        self.skip_bits(n);
77        v
78    }
79
80    /// Show signed bits (two's complement), equivalent to `show_sbits()`.
81    #[inline]
82    pub fn show_sbits(&self, n: usize) -> i32 {
83        debug_assert!(n > 0 && n <= 32);
84        let v = self.show_bits(n) as i32;
85        Self::sign_extend(v, n)
86    }
87
88    /// Get signed bits (two's complement), equivalent to `get_sbits()`.
89    #[inline]
90    pub fn get_sbits(&mut self, n: usize) -> i32 {
91        let v = self.show_sbits(n);
92        self.skip_bits(n);
93        v
94    }
95
96    /// Equivalent to `get_xbits()` for MPEG dc-style signed values.
97    #[inline]
98    pub fn get_xbits(&mut self, n: usize) -> i32 {
99        debug_assert!(n > 0 && n <= 25);
100        let v = self.get_bits(n) as i32;
101        let thresh = 1 << (n - 1);
102        if v < thresh {
103            v - ((1 << n) - 1)
104        } else {
105            v
106        }
107    }
108
109    #[inline]
110    pub fn sign_extend(val: i32, bits: usize) -> i32 {
111        debug_assert!(bits > 0 && bits <= 32);
112        let shift = 32 - bits;
113        (val << shift) >> shift
114    }
115
116    /// Equivalent to `skip_1stop_8data_bits()`.
117    pub fn skip_1stop_8data_bits(&mut self) -> Result<()> {
118        if self.bits_left() <= 0 {
119            return Err(DecodeError::InvalidData("slice extra bits"));
120        }
121        while self.get_bits1() != 0 {
122            self.skip_bits(8);
123            if self.bits_left() <= 0 {
124                return Err(DecodeError::InvalidData("slice extra bits"));
125            }
126        }
127        Ok(())
128    }
129}
