35fn build_table(vlc: &mut Vlc, table_nb_bits: i32, codes: &mut [VlcCode]) -> Result<usize, ()> {
36    if table_nb_bits > 30 {
37        return Err(());
38    }
39    let table_size = 1usize << (table_nb_bits as usize);
40    let table_index = alloc_table(vlc, table_size);
41
42    // first pass
43    let mut i = 0usize;
44    while i < codes.len() {
45        let mut n = codes[i].bits as i32;
46        let mut code = codes[i].code;
47        let symbol = codes[i].symbol;
48
49        if n <= table_nb_bits {
50            let mut j = (code >> (32 - table_nb_bits)) as usize;
51            let nb = 1usize << ((table_nb_bits - n) as usize);
52            for _ in 0..nb {
53                let entry = &mut vlc.table[table_index + j];
54                let bits = entry.len;
55                let oldsym = entry.sym;
56                if (bits != 0 || oldsym != 0) && (bits as i32 != n || oldsym != symbol) {
57                    return Err(());
58                }
59                entry.len = n as i16;
60                entry.sym = symbol;
61                j += 1;
62            }
63            i += 1;
64        } else {
65            // subtable
66            n -= table_nb_bits;
67            let code_prefix = code >> (32 - table_nb_bits);
68            let mut subtable_bits = n;
69
70            codes[i].bits = n as u8;
71            codes[i].code = code << (table_nb_bits as u32);
72
73            let mut k = i + 1;
74            while k < codes.len() {
75                let n2 = codes[k].bits as i32 - table_nb_bits;
76                if n2 <= 0 {
77                    break;
78                }
79                let code2 = codes[k].code;
80                if (code2 >> (32 - table_nb_bits)) != code_prefix {
81                    break;
82                }
83                codes[k].bits = n2 as u8;
84                codes[k].code = code2 << (table_nb_bits as u32);
85                if n2 > subtable_bits {
86                    subtable_bits = n2;
87                }
88                k += 1;
89            }
90
91            if subtable_bits > table_nb_bits {
92                subtable_bits = table_nb_bits;
93            }
94
95            let j = code_prefix as usize;
96            let e = &mut vlc.table[table_index + j];
97            e.len = -(subtable_bits as i16);
98
99            // recurse
100            let sub_index = build_table(vlc, subtable_bits, &mut codes[i..k])?;
101            // reload entry
102            vlc.table[table_index + j].sym = sub_index as i16;
103
104            i = k;
105        }
106    }
107
108    // finalize: mark illegal
109    for t in &mut vlc.table[table_index..table_index + table_size] {
110        if t.len == 0 {
111            t.sym = -1;
112        }
113    }
114
115    Ok(table_index)
116}

118impl Vlc {
119    pub fn init_sparse(nb_bits: i32, bits: &[u8], codes: &[u16], symbols: Option<&[i16]>) -> Self {
120        // Build a canonical list of (code,len,symbol) entries and sort by the
121        // MSB-aligned code.
122        //
123        // IMPORTANT: VLC table construction relies on codes being grouped by
124        // prefix. Any ordering hack (e.g. splitting by length or sorting by a
125        // shifted key like `code >> 1`) will corrupt the table and cause
126        // widespread "invalid data: ac" failures.
127        let mut buf: Vec<VlcCode> = Vec::with_capacity(bits.len());
128
129        #[inline]
130        fn push_code(
131            buf: &mut Vec<VlcCode>,
132            i: usize,
133            bits: &[u8],
134            codes: &[u16],
135            symbols: Option<&[i16]>,
136        ) {
137            let len = bits[i] as i32;
138            if len <= 0 {
139                return;
140            }
141            let mut code = codes[i] as u32;
142            // validate
143            if len > 32 {
144                return;
145            }
146            if code >= (1u32 << len) {
147                return;
148            }
149            code <<= 32 - len;
150            let sym = symbols.map(|s| s[i]).unwrap_or(i as i16);
151            buf.push(VlcCode { bits: len as u8, symbol: sym, code });
152        }
153
154        for i in 0..bits.len() {
155            push_code(&mut buf, i, bits, codes, symbols);
156        }
157
158        // Sort by full MSB-aligned code (NOT shifted) to avoid collisions.
159        buf.sort_by(|a, b| a.code.cmp(&b.code));
160
161        let mut vlc = Vlc { bits: nb_bits, table: Vec::new() };
162        let _ = build_table(&mut vlc, nb_bits, &mut buf).unwrap();
163        vlc
164    }
165}

167/// Equivalent to `get_vlc2()`.
168#[inline]
169pub fn get_vlc2(gb: &mut GetBits<'_>, table: &[VlcElem], bits: i32, max_depth: i32) -> i32 {
170    let mut index = gb.show_bits(bits as usize) as usize;
171    let mut code = table[index].sym as i32;
172    let mut n = table[index].len as i32;
173
174    if max_depth > 1 && n < 0 {
175        gb.skip_bits(bits as usize);
176        let mut nb_bits = -n;
177        index = gb.show_bits(nb_bits as usize) as usize + (code as usize);
178        code = table[index].sym as i32;
179        n = table[index].len as i32;
180        if max_depth > 2 && n < 0 {
181            gb.skip_bits(nb_bits as usize);
182            nb_bits = -n;
183            index = gb.show_bits(nb_bits as usize) as usize + (code as usize);
184            code = table[index].sym as i32;
185            n = table[index].len as i32;
186        }
187    }
188
189    if n > 0 {
190        gb.skip_bits(n as usize);
191    }
192    code
193}

195/// Equivalent to `GET_RL_VLC` macro for run/level VLC tables.
196#[inline]
197pub fn get_rl_vlc(gb: &mut GetBits<'_>, table: &[RlVlcElem], bits: i32, max_depth: i32) -> (i16, u8) {
198    let mut index = gb.show_bits(bits as usize) as usize;
199    let mut level = table[index].level;
200    let mut n = table[index].len8 as i32;
201
202    if max_depth > 1 && n < 0 {
203        gb.skip_bits(bits as usize);
204        let mut nb_bits = -n;
205        index = gb.show_bits(nb_bits as usize) as usize + (level as usize);
206        level = table[index].level;
207        n = table[index].len8 as i32;
208        if max_depth > 2 && n < 0 {
209            gb.skip_bits(nb_bits as usize);
210            nb_bits = -n;
211            index = gb.show_bits(nb_bits as usize) as usize + (level as usize);
212            level = table[index].level;
213            n = table[index].len8 as i32;
214        }
215    }
216
217    let run = table[index].run;
218    if n > 0 {
219        gb.skip_bits(n as usize);
220    }
