9enum CondKind {
10    NonZero,
11    Zero,
12    AlwaysTrue,
13    AlwaysFalse,
14}

16fn parse_cond(cond: &str) -> CondKind {
17    let c = cond.trim();
18    if c == "true" {
19        return CondKind::AlwaysTrue;
20    }
21    if c == "false" || c == "nil" {
22        return CondKind::AlwaysFalse;
23    }
24
25    // Accept optional parentheses.
26    let c = c.trim_start_matches('(').trim_end_matches(')').trim();
27
28    // Typical IR conditions.
29    // - Sx ~= 0  => NonZero
30    // - Sx == 0  => Zero
31    // - Sx       => NonZero
32    let re_ne0 = Regex::new(r"^S\d+\s*~=\s*0$" ).unwrap();
33    let re_eq0 = Regex::new(r"^S\d+\s*==\s*0$" ).unwrap();
34    let re_s = Regex::new(r"^S\d+$" ).unwrap();
35
36    if re_ne0.is_match(c) {
37        return CondKind::NonZero;
38    }
39    if re_eq0.is_match(c) {
40        return CondKind::Zero;
41    }
42    if re_s.is_match(c) {
43        return CondKind::NonZero;
44    }
45
46    // Fallback: treat as non-zero; this keeps the compiler permissive
47    // for decompiler-emitted conditions like "(S0 ~= 0)".
48    CondKind::NonZero
49}

51fn slot_to_stack_idx(var: &str, args_count: i8) -> Result<i8> {
52    let re_a = Regex::new(r"^a(\d+)$").unwrap();
53    let re_l = Regex::new(r"^l(\d+)$").unwrap();
54
55    if let Some(c) = re_a.captures(var) {
56        let v: i16 = c.get(1).unwrap().as_str().parse()?;
57        return Ok(i8::try_from(v).map_err(|_| anyhow!("stack index out of i8"))?);
58    }
59    if let Some(c) = re_l.captures(var) {
60        let v: i16 = c.get(1).unwrap().as_str().parse()?;
61        let idx: i16 = i16::from(args_count) + v;
62        return Ok(i8::try_from(idx).map_err(|_| anyhow!("stack index out of i8"))?);
63    }
64    bail!("not a frame slot: {var}")
65}

67fn push_int(v: i64) -> Result<OpKind> {
68    if v < i64::from(i32::MIN) || v > i64::from(i32::MAX) {
69        bail!("integer out of i32 range: {v}");
70    }
71    if v >= -128 && v <= 127 {
72        return Ok(OpKind::PushI8(v as i8));
73    }
74    if v >= -32768 && v <= 32767 {
75        return Ok(OpKind::PushI16(v as i16));
76    }
77    Ok(OpKind::PushI32(v as i32))
78}

80fn lua_unescape_string(lit: &str) -> String {
81    // Limited support: \\ and \" and \n \r \t.
82    let mut out = String::new();
83    let mut chars = lit.chars();
84    while let Some(ch) = chars.next() {
85        if ch != '\\' {
86            out.push(ch);
87            continue;
88        }
89        match chars.next() {
90            Some('n') => out.push('\n'),
91            Some('r') => out.push('\r'),
92            Some('t') => out.push('\t'),
93            Some('\\') => out.push('\\'),
94            Some('"') => out.push('"'),
95            Some(other) => out.push(other),
96            None => break,
97        }
98    }
99    out
100}

102fn compile_simple_stmt(
103    stmt: &str,
104    args_count: i8,
105    meta: &Meta,
106    user_fns: &HashSet<String>,
107    out: &mut Vec<Item>,
108) -> Result<()> {
109    let s = stmt.trim();
110
111    if s.is_empty() {
112        return Ok(());
113    }
114
115    // Some decompiler variants include explicit literal writes to __ret (e.g. `__ret = nil`).
116    // The HCB VM's return register is implicitly produced by call/syscall, so these statements
117    // are non-semantic and should be ignored.
118    {let re = Regex::new(r#"^__ret\s*=\s*(nil|true|false|-?\d+(?:\.\d+)?|"(?:\\.|[^"])*")\s*$"#).unwrap();
119        
120        if re.is_match(s) {
121            return Ok(());
122        }
123    }
124
125    // __ret = foo(...)
126    {
127        let re = Regex::new(r"^__ret\s*=\s*(\w+)\s*\((.*)\)\s*$").unwrap();
128        if let Some(c) = re.captures(s) {
129            let name = c.get(1).unwrap().as_str();
130            let args_s = c.get(2).unwrap().as_str().trim();
131            let argc = if args_s.is_empty() {
132                0usize
133            } else {
134                args_s.split(',').map(|x| x.trim()).filter(|x| !x.is_empty()).count()
135            };
136
137            if let Some(sid) = meta.syscall_id_by_name(name) {
138                out.push(Item::Op(OpKind::Syscall { id: sid }));
139                let _ = argc; // argc is for human readability; opcode has only id.
140                return Ok(());
141            }
142
143            if name.starts_with("f_0x") || user_fns.contains(name) {
144                out.push(Item::Op(OpKind::CallFn {
145                    name: name.to_string(),
146                }));
147                let _ = argc;
148                return Ok(());
149            }
150
151            bail!("unknown callee: {name}");
152        }
153    }
154
155    // foo(...) as statement
156    {
157        let re = Regex::new(r"^(\w+)\s*\((.*)\)\s*$").unwrap();
158        if let Some(c) = re.captures(s) {
159            let name = c.get(1).unwrap().as_str();
160            let args_s = c.get(2).unwrap().as_str().trim();
161            let _argc = if args_s.is_empty() {
162                0usize
163            } else {
164                args_s.split(',').map(|x| x.trim()).filter(|x| !x.is_empty()).count()
165            };
166
167            if let Some(sid) = meta.syscall_id_by_name(name) {
168                out.push(Item::Op(OpKind::Syscall { id: sid }));
169                return Ok(());
170            }
171            if name.starts_with("f_0x") || user_fns.contains(name) {
172                out.push(Item::Op(OpKind::CallFn {
173                    name: name.to_string(),
174                }));
175                return Ok(());
176            }
177        }
178    }
179
180    // Sx = ...
181    {
182        let re = Regex::new(r"^S\d+\s*=\s*(.+)$").unwrap();
183        if let Some(c) = re.captures(s) {
184            let rhs = c.get(1).unwrap().as_str().trim();
185
186            // Literals
187            if rhs == "nil" {
188                out.push(Item::Op(OpKind::PushNil));
189                return Ok(());
190            }
191            if rhs == "true" {
192                out.push(Item::Op(OpKind::PushTrue));
193                return Ok(());
194            }
195            if rhs == "__ret" {
196                out.push(Item::Op(OpKind::PushReturn));
197                return Ok(());
198            }
199
200            if let Ok(v) = rhs.parse::<i64>() {
201                out.push(Item::Op(push_int(v)?));
202                return Ok(());
203            }
204
205            if let Ok(v) = rhs.parse::<f32>() {
206                // Distinguish integer vs float by presence of '.' or exponent.
207                if rhs.contains('.') || rhs.contains('e') || rhs.contains('E') {
208                    out.push(Item::Op(OpKind::PushF32(v)));
209                    return Ok(());
210                }
211            }
212
213            {
214                let re = Regex::new(r#"^"(.*)"$"# ).unwrap();
215                if let Some(mm) = re.captures(rhs) {
216                    let lit = mm.get(1).unwrap().as_str();
217                    out.push(Item::Op(OpKind::PushString(lua_unescape_string(lit))));
218                    return Ok(());
219                }
220            }
221
222            // push_top pattern: Sx = Sy
223            {
224                let re = Regex::new(r"^S\d+$").unwrap();
225                if re.is_match(rhs) {
226                    out.push(Item::Op(OpKind::PushTop));
227                    return Ok(());
228                }
229            }
230
231            // push_global
232            {
233                let re = Regex::new(r"^G\[(\d+)\]$" ).unwrap();
234                if let Some(mm) = re.captures(rhs) {
235                    let idx: u16 = mm.get(1).unwrap().as_str().parse()?;
236                    out.push(Item::Op(OpKind::PushGlobal(idx)));
237                    return Ok(());
238                }
239            }
240
241            // push_stack (frame slots)
242            {
243                let re = Regex::new(r"^(a\d+|l\d+)$").unwrap();
244                if re.is_match(rhs) {
245                    let idx = slot_to_stack_idx(rhs, args_count)?;
246                    out.push(Item::Op(OpKind::PushStack(idx)));
247                    return Ok(());
248                }
249            }
250
251            // table reads
252            {
253                let re = Regex::new(r"^GT\[(\d+)\]\[S\d+\]$" ).unwrap();
254                if let Some(mm) = re.captures(rhs) {
255                    let idx: u16 = mm.get(1).unwrap().as_str().parse()?;
256                    out.push(Item::Op(OpKind::PushGlobalTable(idx)));
257                    return Ok(());
258                }
259            }
260            {
261                let re = Regex::new(r"^LT\[(-?\d+)\]\[S\d+\]$" ).unwrap();
262                if let Some(mm) = re.captures(rhs) {
263                    let idx: i8 = mm.get(1).unwrap().as_str().parse()?;
264                    out.push(Item::Op(OpKind::PushLocalTable(idx)));
265                    return Ok(());
266                }
267            }
268
269            // unary neg: -Sx
270            {
271                let re = Regex::new(r"^-S\d+$").unwrap();
272                if re.is_match(rhs) {
273                    out.push(Item::Op(OpKind::Neg));
274                    return Ok(());
275                }
276            }
277
278            // arithmetic: Sx = Sa + Sb
279            {
280                let re = Regex::new(r"^S\d+\s*([+\-*/%])\s*S\d+$" ).unwrap();
281                if let Some(mm) = re.captures(rhs) {
282                    let op = mm.get(1).unwrap().as_str();
283                    let k = match op {
284                        "+" => OpKind::Add,
285                        "-" => OpKind::Sub,
286                        "*" => OpKind::Mul,
287                        "/" => OpKind::Div,
288                        "%" => OpKind::Mod,
289                        _ => bail!("unknown arithmetic op: {op}"),
290                    };
291                    out.push(Item::Op(k));
292                    return Ok(());
293                }
294            }
295
296            // bit_test: (Sa & Sb) ~= 0
297            if rhs.contains('&') && rhs.contains("~= 0") {
298                out.push(Item::Op(OpKind::BitTest));
299                return Ok(());
300            }
301
302            // and/or (decompiler shape)
303            if rhs.contains(" and ") && rhs.contains("~= nil") {
304                out.push(Item::Op(OpKind::And));
305                return Ok(());
306            }
307            if rhs.contains(" or ") && rhs.contains("~= nil") {
308                out.push(Item::Op(OpKind::Or));
309                return Ok(());
310            }
311
312            // comparisons
313            {
314                let re = Regex::new(r"^\(S\d+\s*(==|~=|>|<=|<|>=)\s*S\d+\)$" ).unwrap();
315                if let Some(mm) = re.captures(rhs) {
316                    let cop = mm.get(1).unwrap().as_str();
317                    let k = match cop {
318                        "==" => OpKind::SetE,
319                        "~=" => OpKind::SetNe,
320                        ">" => OpKind::SetG,
321                        "<=" => OpKind::SetLe,
322                        "<" => OpKind::SetL,
323                        ">=" => OpKind::SetGe,
324                        _ => bail!("unknown compare op: {cop}"),
325                    };
326                    out.push(Item::Op(k));
327                    return Ok(());
328                }
329            }
330
331            // RHS call: Sx = foo(...)
332            {
333                let re = Regex::new(r"^(\w+)\s*\((.*)\)\s*$" ).unwrap();
334                if let Some(mm) = re.captures(rhs) {
335                    let name = mm.get(1).unwrap().as_str();
336                    let args_s = mm.get(2).unwrap().as_str().trim();
337                    let _argc = if args_s.is_empty() {
338                        0usize
339                    } else {
340                        args_s.split(',').map(|x| x.trim()).filter(|x| !x.is_empty()).count()
341                    };
342
343                    if let Some(sid) = meta.syscall_id_by_name(name) {
344                        out.push(Item::Op(OpKind::Syscall { id: sid }));
345                        out.push(Item::Op(OpKind::PushReturn));
346                        return Ok(());
347                    }
348                    if name.starts_with("f_0x") || user_fns.contains(name) {
349                        out.push(Item::Op(OpKind::CallFn {
350                            name: name.to_string(),
351                        }));
352                        out.push(Item::Op(OpKind::PushReturn));
353                        return Ok(());
354                    }
355                    bail!("unknown callee: {name}");
356                }
357            }
358
359            bail!("unsupported S assignment: {s}");
360        }
361    }
362
363    // G[idx] = Sx
364    {
365        let re = Regex::new(r"^G\[(\d+)\]\s*=\s*S\d+\s*$" ).unwrap();
366        if let Some(c) = re.captures(s) {
367            let idx: u16 = c.get(1).unwrap().as_str().parse()?;
368            out.push(Item::Op(OpKind::PopGlobal(idx)));
369            return Ok(());
370        }
371    }
372
373    // G[idx] = __ret
374    {
375        let re = Regex::new(r"^G\[(\d+)\]\s*=\s*__ret\s*$" ).unwrap();
376        if let Some(c) = re.captures(s) {
377            let idx: u16 = c.get(1).unwrap().as_str().parse()?;
378            out.push(Item::Op(OpKind::PushReturn));
379            out.push(Item::Op(OpKind::PopGlobal(idx)));
380            return Ok(());
381        }
382    }
383
384    // aN/lN = Sx
385    {
386        let re = Regex::new(r"^(a\d+|l\d+)\s*=\s*S\d+\s*$" ).unwrap();
387        if let Some(c) = re.captures(s) {
388            let var = c.get(1).unwrap().as_str();
389            let idx = slot_to_stack_idx(var, args_count)?;
390            out.push(Item::Op(OpKind::PopStack(idx)));
391            return Ok(());
392        }
393    }
394
395    // aN/lN = __ret
396    {
397        let re = Regex::new(r"^(a\d+|l\d+)\s*=\s*__ret\s*$" ).unwrap();
398        if let Some(c) = re.captures(s) {
399            let var = c.get(1).unwrap().as_str();
400            let idx = slot_to_stack_idx(var, args_count)?;
401            out.push(Item::Op(OpKind::PushReturn));
402            out.push(Item::Op(OpKind::PopStack(idx)));
403            return Ok(());
404        }
405    }
406
407    // table store: GT[i][Skey] = Sval
408    {
409        let re = Regex::new(r"^GT\[(\d+)\]\[S\d+\]\s*=\s*S\d+\s*$" ).unwrap();
410        if let Some(c) = re.captures(s) {
411            let idx: u16 = c.get(1).unwrap().as_str().parse()?;
412            out.push(Item::Op(OpKind::PopGlobalTable(idx)));
413            return Ok(());
414        }
415    }
416
417    // table store: LT[i][Skey] = Sval
418    {
419        let re = Regex::new(r"^LT\[(-?\d+)\]\[S\d+\]\s*=\s*S\d+\s*$" ).unwrap();
420        if let Some(c) = re.captures(s) {
421            let idx: i8 = c.get(1).unwrap().as_str().parse()?;
422            out.push(Item::Op(OpKind::PopLocalTable(idx)));
423            return Ok(());
424        }
425    }
426
427    bail!("unsupported statement: {s}");
428}

430struct LabelGen {
431    prefix: String,
432    n: u32,
433}
434
435impl LabelGen {
436    fn new(prefix: impl Into<String>) -> Self {
437        Self {
438            prefix: prefix.into(),
439            n: 0,
440        }
441    }
442
443    fn fresh(&mut self, kind: &str) -> String {
444        let id = self.n;
445        self.n += 1;
446        format!("{}:{}:{}", self.prefix, kind, id)
447    }
448}

450fn compile_stmts(
451    stmts: &[Stmt],
452    args_count: i8,
453    meta: &Meta,
454    user_fns: &HashSet<String>,
455    out: &mut Vec<Item>,
456    lg: &mut LabelGen,
457    break_stack: &mut Vec<String>,
458) -> Result<()> {
459    for st in stmts {
460        match st {
461            Stmt::Simple(s) => compile_simple_stmt(s, args_count, meta, user_fns, out)?,
462            Stmt::Return(None) => out.push(Item::Op(OpKind::Ret)),
463            Stmt::Return(Some(expr)) => {
464                // Return value is expected to already be on stack (as Sx), following the IR convention.
465                // We support limited literals here for convenience.
466                let e = expr.trim();
467                if Regex::new(r"^S\d+$").unwrap().is_match(e) {
468                    out.push(Item::Op(OpKind::Retv));
469                } else if e == "nil" {
470                    out.push(Item::Op(OpKind::PushNil));
471                    out.push(Item::Op(OpKind::Retv));
472                } else if e == "true" {
473                    out.push(Item::Op(OpKind::PushTrue));
474                    out.push(Item::Op(OpKind::Retv));
475                } else if let Ok(v) = e.parse::<i64>() {
476                    out.push(Item::Op(push_int(v)?));
477                    out.push(Item::Op(OpKind::Retv));
478                } else {
479                    bail!("unsupported return expr: {e}");
480                }
481            }
482            Stmt::Break => {
483                let tgt = break_stack
484                    .last()
485                    .ok_or_else(|| anyhow!("break outside of loop"))?
486                    .clone();
487                out.push(Item::Op(OpKind::JmpLabel { label: tgt }));
488            }
489            Stmt::If { arms, else_arm } => {
490                let end_lbl = lg.fresh("if_end");
491                let mut next_lbl = lg.fresh("if_next");
492
493                for (idx, (cond, body)) in arms.iter().enumerate() {
494                    let ck = parse_cond(cond);
495                    let body_lbl = lg.fresh(&format!("if_body_{idx}"));
496                    let after_lbl = if idx + 1 == arms.len() {
497                        // Fallthrough to else or end.
498                        next_lbl.clone()
499                    } else {
500                        lg.fresh(&format!("if_next_{idx}"))
501                    };
502
503                    match ck {
504                        CondKind::AlwaysTrue => {
505                            // Compile body and jump to end; ignore remaining arms.
506                            compile_stmts(body, args_count, meta, user_fns, out, lg, break_stack)?;
507                            out.push(Item::Op(OpKind::JmpLabel { label: end_lbl.clone() }));
508                            next_lbl = after_lbl;
509                            break;
510                        }
511                        CondKind::AlwaysFalse => {
512                            // Skip to after_lbl.
513                            out.push(Item::Op(OpKind::JmpLabel { label: after_lbl.clone() }));
514                        }
515                        CondKind::NonZero => {
516                            out.push(Item::Op(OpKind::JzLabel {
517                                label: after_lbl.clone(),
518                            }));
519                            compile_stmts(body, args_count, meta, user_fns, out, lg, break_stack)?;
520                            out.push(Item::Op(OpKind::JmpLabel { label: end_lbl.clone() }));
521                        }
522                        CondKind::Zero => {
523                            out.push(Item::Op(OpKind::JzLabel {
524                                label: body_lbl.clone(),
525                            }));
526                            out.push(Item::Op(OpKind::JmpLabel {
527                                label: after_lbl.clone(),
528                            }));
529                            out.push(Item::Label(Label::new(body_lbl)));
530                            compile_stmts(body, args_count, meta, user_fns, out, lg, break_stack)?;
531                            out.push(Item::Op(OpKind::JmpLabel { label: end_lbl.clone() }));
532                        }
533                    }
534
535                    out.push(Item::Label(Label::new(after_lbl.clone())));
536                    next_lbl = after_lbl;
537                }
538
539                if let Some(eb) = else_arm {
540                    compile_stmts(eb, args_count, meta, user_fns, out, lg, break_stack)?;
541                }
542
543                out.push(Item::Label(Label::new(end_lbl)));
544            }
545            Stmt::While { cond, body } => {
546                let head = lg.fresh("while_head");
547                let end = lg.fresh("while_end");
548                let body_lbl = lg.fresh("while_body");
549
550                out.push(Item::Label(Label::new(head.clone())));
551
552                break_stack.push(end.clone());
553
554                match parse_cond(cond) {
555                    CondKind::AlwaysTrue => {
556                        // No condition check.
557                        compile_stmts(body, args_count, meta, user_fns, out, lg, break_stack)?;
558                        out.push(Item::Op(OpKind::JmpLabel { label: head }));
559                    }
560                    CondKind::AlwaysFalse => {
561                        out.push(Item::Op(OpKind::JmpLabel { label: end.clone() }));
562                    }
563                    CondKind::NonZero => {
564                        out.push(Item::Op(OpKind::JzLabel { label: end.clone() }));
565                        compile_stmts(body, args_count, meta, user_fns, out, lg, break_stack)?;
566                        out.push(Item::Op(OpKind::JmpLabel { label: head }));
567                    }
568                    CondKind::Zero => {
569                        out.push(Item::Op(OpKind::JzLabel { label: body_lbl.clone() }));
570                        out.push(Item::Op(OpKind::JmpLabel { label: end.clone() }));
571                        out.push(Item::Label(Label::new(body_lbl)));
572                        compile_stmts(body, args_count, meta, user_fns, out, lg, break_stack)?;
573                        out.push(Item::Op(OpKind::JmpLabel { label: head }));
574                    }
575                }
576
577                break_stack.pop();
578
579                out.push(Item::Label(Label::new(end)));
580            }
581        }
582    }
583    Ok(())
584}

590fn looks_like_pc_dispatcher(raw: &[String]) -> bool {
591    // Typical decompiler pattern:
592    //   local __pc = 0
593    //   while true do
594    //     if __pc == 0 then
595    //       ...
596    //     elseif __pc == 1 then
597    //       ...
598    //     else
599    //       return
600    //     end
601    //   end
602    let mut saw_pc = false;
603    let mut saw_case = false;
604    let mut saw_while_true = false;
605    for ln in raw {
606        let t = ln.trim();
607        if t.contains("__pc") {
608            saw_pc = true;
609        }
610        if t == "while true do" {
611            saw_while_true = true;
612        }
613        if t.starts_with("if __pc ==") || t.starts_with("elseif __pc ==") {
614            saw_case = true;
615        }
616    }
617    saw_pc && saw_while_true && saw_case
618}

620fn is_comment_or_empty_line(t: &str) -> bool {
621    let tt = t.trim();

652fn bb_label(fn_name: &str, pc: u32) -> String {

656fn parse_entry_pc(body: &[String]) -> u32 {
657    let re = Regex::new(r"^(?:local\s+)?__pc\s*=\s*(\d+)\s*$").unwrap();
658    for ln in body {
659        let t = ln.trim();
660        if let Some(c) = re.captures(t) {
661            if let Ok(v) = c.get(1).unwrap().as_str().parse::<u32>() {
662                return v;
663            }
664        }
665        if t == "while true do" {
666            break;
667        }
668    }
669    0
670}

672fn collect_case_body(body: &[String], mut i: usize, re_case: &Regex) -> (Vec<String>, usize) {
673    let mut out: Vec<String> = Vec::new();
674    let mut nest: i32 = 0;
675
676    while i < body.len() {
677        let t = body[i].trim();
678        if nest == 0 {
679            if re_case.is_match(t) || t == "else" {
680                break;
681            }
682        }
683
684        out.push(body[i].clone());
685
686        if is_if_start_line(t) || is_while_start_line(t) || is_for_start_line(t) || is_repeat_start_line(t) {
687            nest += 1;
688        } else if is_end_line(t) {
689            nest -= 1;
690        } else if is_until_line(t) {
691            nest -= 1;
692        }
693
694        i += 1;
695    }
696
697    (out, i)
698}

700fn compile_pc_case(
701    pc: u32,
702    lines: &[String],
703    fn_name: &str,
704    args_count: i8,
705    meta: &Meta,
706    user_fns: &HashSet<String>,
707    out: &mut Vec<Item>,
708) -> Result<()> {
709    out.push(Item::Label(Label::new(bb_label(fn_name, pc))));
710
711    let re_pc_set = Regex::new(r"^__pc\s*=\s*(\d+)\s*$").unwrap();
712    let re_term_if = Regex::new(r"^if\s+S\d+\s*(==|~=)\s*0\s+then\s*$").unwrap();
713
714    let mut i = 0usize;
715    while i < lines.len() {
716        let mut t = lines[i].trim().to_string();
717        if is_comment_or_empty_line(&t) {
718            i += 1;
719            continue;
720        }
721
722        // Drop pure local declarations in a case.
723        if t.starts_with("local ") && !t.contains('=') {
724            i += 1;
725            continue;
726        }
727        if let Some(rest) = t.strip_prefix("local ") {
728            t = rest.trim().to_string();
729        }
730
731        // return / return <expr>
732        if t == "return" {
733            out.push(Item::Op(OpKind::Ret));
734            return Ok(());
735        }
736
737        // Some decompiler variants emit `return Sx` for single-value returns.
738        // In this IR, `Sx` is expected to be the current top-of-stack, so we can
739        // directly emit `retv` (pop one value and return it).
740        if let Some(rest) = t.strip_prefix("return ") {
741            let e = rest.trim();
742
743            if Regex::new(r"^S\d+$").unwrap().is_match(e) {
744                out.push(Item::Op(OpKind::Retv));
745                return Ok(());
746            }
747
748            // Literal returns (convenience)
749            if e == "nil" || e == "false" {
750                out.push(Item::Op(OpKind::PushNil));
751                out.push(Item::Op(OpKind::Retv));
752                return Ok(());
753            }
754            if e == "true" {
755                out.push(Item::Op(OpKind::PushTrue));
756                out.push(Item::Op(OpKind::Retv));
757                return Ok(());
758            }
759            if let Ok(v) = e.parse::<i64>() {
760                out.push(Item::Op(push_int(v)?));
761                out.push(Item::Op(OpKind::Retv));
762                return Ok(());
763            }
764            if let Ok(vf) = e.parse::<f32>() {
765                // Distinguish integer vs float by presence of '.' or exponent.
766                if e.contains('.') || e.contains('e') || e.contains('E') {
767                    out.push(Item::Op(OpKind::PushF32(vf)));
768                    out.push(Item::Op(OpKind::Retv));
769                    return Ok(());
770                }
771            }
772            if let Some(mm) = Regex::new(r#"^\"(.*)\"$"#).unwrap().captures(e) {
773                let lit = mm.get(1).unwrap().as_str();
774                out.push(Item::Op(OpKind::PushString(lua_unescape_string(lit))));
775                out.push(Item::Op(OpKind::Retv));
776                return Ok(());
777            }
778
779            bail!("unsupported return expr in pc bb {pc}: {e}");
780        }
781
782        // Unconditional jump via __pc = N
783        if let Some(c) = re_pc_set.captures(&t) {
784            let target: u32 = c.get(1).unwrap().as_str().parse()?;
785            out.push(Item::Op(OpKind::JmpLabel {
786                label: bb_label(fn_name, target),
787            }));
788            return Ok(());
789        }
790
791        // Terminator: if Sx (==|~=) 0 then __pc=A else __pc=B end
792        if let Some(c) = re_term_if.captures(&t) {
793            let op = c.get(1).unwrap().as_str(); // == or ~=
794            // then: __pc = A
795            let mut j = i + 1;
796            while j < lines.len() && is_comment_or_empty_line(lines[j].trim()) {
797                j += 1;
798            }
799            if j >= lines.len() {
800                bail!("unterminated pc-if in bb {pc}");
801            }
802            let then_line = lines[j].trim();
803            let then_pc: u32 = re_pc_set
804                .captures(then_line)
805                .ok_or_else(|| anyhow!("pc-if then arm must set __pc in bb {pc}"))?
806                .get(1)
807                .unwrap()
808                .as_str()
809                .parse()?;
810
811            j += 1;
812            while j < lines.len() && is_comment_or_empty_line(lines[j].trim()) {
813                j += 1;
814            }
815            if j >= lines.len() || lines[j].trim() != "else" {
816                bail!("pc-if missing else in bb {pc}");
817            }
818
819            j += 1;
820            while j < lines.len() && is_comment_or_empty_line(lines[j].trim()) {
821                j += 1;
822            }
823            if j >= lines.len() {
824                bail!("pc-if missing else pc assignment in bb {pc}");
825            }
826            let else_line = lines[j].trim();
827            let else_pc: u32 = re_pc_set
828                .captures(else_line)
829                .ok_or_else(|| anyhow!("pc-if else arm must set __pc in bb {pc}"))?
830                .get(1)
831                .unwrap()
832                .as_str()
833                .parse()?;
834
835            j += 1;
836            while j < lines.len() && is_comment_or_empty_line(lines[j].trim()) {
837                j += 1;
838            }
839            if j >= lines.len() || lines[j].trim() != "end" {
840                bail!("pc-if missing end in bb {pc}");
841            }
842
843            // Encode with the VM's "jz" (jump on zero).
844            // if Sx == 0: jz then_pc else jmp else_pc
845            // if Sx ~= 0: jz else_pc else jmp then_pc
846            let (zero_target, nonzero_target) = if op == "==" {
847                (then_pc, else_pc)
848            } else {
849                (else_pc, then_pc)
850            };
851            out.push(Item::Op(OpKind::JzLabel {
852                label: bb_label(fn_name, zero_target),
853            }));
854            out.push(Item::Op(OpKind::JmpLabel {
855                label: bb_label(fn_name, nonzero_target),
856            }));
857            return Ok(());
858        }
859
860        // Normal statement inside bb.
861        compile_simple_stmt(&t, args_count, meta, user_fns, out)?;
862        i += 1;
863    }
864
865    // Conservative fallback.
866    out.push(Item::Op(OpKind::Ret));
867    Ok(())
868}
869
870fn compile_pc_dispatcher_function(
871    f: &Function,
872    meta: &Meta,
873    user_fns: &HashSet<String>,
874    out: &mut Vec<Item>,
875) -> Result<()> {
876    if f.raw.len() < 2 {
877        bail!("function {}: too short", f.name);
878    }
879    let body: Vec<String> = f.raw[1..f.raw.len() - 1].to_vec();
880
881    let entry_pc = parse_entry_pc(&body);
882
883    let re_case = Regex::new(r"^(if|elseif)\s+__pc\s*==\s*(\d+)\s+then\s*$").unwrap();
884
885    // Find the first dispatcher case.
886    let mut i = 0usize;
887    while i < body.len() {
888        if re_case.is_match(body[i].trim()) {
889            break;
890        }
891        i += 1;
892    }
893    if i >= body.len() {
894        bail!("function {}: pc-dispatcher header not found", f.name);
895    }
896
897    let mut cases: Vec<(u32, Vec<String>)> = Vec::new();
898    while i < body.len() {
899        let t = body[i].trim();
900        if t == "else" {
901            break;
902        }
903        if let Some(c) = re_case.captures(t) {
904            let pc: u32 = c.get(2).unwrap().as_str().parse()?;
905            i += 1;
906            let (case_lines, next_i) = collect_case_body(&body, i, &re_case);
907            cases.push((pc, case_lines));
908            i = next_i;
909            continue;
910        }
911        i += 1;
912    }
913
914    if cases.is_empty() {
915        bail!("function {}: no pc-dispatcher cases found", f.name);
916    }
917
918    // Emit cases with the entry case first (avoid an extra jmp at function start).
919    if let Some(pos) = cases.iter().position(|(pc, _)| *pc == entry_pc) {
920        if pos != 0 {
921            let entry = cases.remove(pos);
922            cases.insert(0, entry);
923        }
924    }
925
926    for (pc, lines) in cases {
927        compile_pc_case(pc, &lines, &f.name, f.args_count, meta, user_fns, out)?;
928    }
929
930    Ok(())
931}
932
933pub fn compile_program(meta: &Meta, funs: &[Function]) -> Result<Vec<Item>> {
934    let mut items: Vec<Item> = Vec::new();
935
936    let user_fns: HashSet<String> = funs.iter().map(|f| f.name.clone()).collect();
937
938    for f in funs {
939        items.push(Item::Label(Label::new(format!("fn:{}", f.name))));
940        items.push(Item::Op(OpKind::InitStack {
941            args: f.args_count,
942            locals: f.locals_count,
943        }));
944
945        if looks_like_pc_dispatcher(&f.raw) {
946            compile_pc_dispatcher_function(f, meta, &user_fns, &mut items)?;
947        } else {
948            let mut lg = LabelGen::new(format!("fn:{}", f.name));
949            let mut break_stack: Vec<String> = Vec::new();
950            compile_stmts(
951                &f.body,
952                f.args_count,
953                meta,
954                &user_fns,
955                &mut items,
956                &mut lg,
957                &mut break_stack,
958            )?;
959        }
960
961        // Ensure function ends with a ret. If the source omitted explicit return, add Ret.
962        if !matches!(items.last(), Some(Item::Op(OpKind::Ret | OpKind::Retv))) {
963            items.push(Item::Op(OpKind::Ret));
964        }
965    }
966
967    Ok(items)
