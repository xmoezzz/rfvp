1mod asm;
2mod compile;
3mod ir;
4mod lua;
5mod meta;
6
7use anyhow::{anyhow, Context, Result};
8use clap::Parser;
9use std::fs;
10use std::path::PathBuf;
11
12#[derive(Parser, Debug)]
13#[command(name = "lua2hcb")]
14#[command(about = "Compile a constrained Lua 5.3 subset (decompiler IR style) back to HCB.")]
15struct Cli {
16    #[arg(long)]
17    meta: PathBuf,
18
19    #[arg(long)]
20    lua: PathBuf,
21
22    #[arg(short, long)]
23    out: PathBuf,
24}
25
26fn main() -> Result<()> {
27    let cli = Cli::parse();
28
29    let meta = meta::load_meta(&cli.meta)
30        .with_context(|| format!("loading meta: {}", cli.meta.display()))?;
31
32    let funs = lua::parse_lua(&cli.lua).with_context(|| "parsing lua functions")?;
33    if funs.is_empty() {
34        return Err(anyhow!("no functions found in lua input"));
35    }
36
37    let items = compile::compile_program(&meta, &funs).with_context(|| "compiling")?;
38    let (code, labels) = asm::assemble(&meta, &items).with_context(|| "assembling")?;
39
40    let entry_lbl = "fn:entry_point".to_string();
41    let entry_point = labels
42        .get(&entry_lbl)
43        .copied()
44        .ok_or_else(|| anyhow!("missing entry function: entry_point"))?;
45
46    let sysdesc = asm::build_sysdesc(&meta, entry_point).with_context(|| "building sysdesc")?;
47
48    let sys_desc_offset: u32 =
49        4 + u32::try_from(code.len()).map_err(|_| anyhow!("code too large"))?;
50
51    let mut out = Vec::new();
52    out.extend_from_slice(&sys_desc_offset.to_le_bytes());
53    out.extend_from_slice(&code);
54    out.extend_from_slice(&sysdesc);
55
56    fs::write(&cli.out, out).with_context(|| format!("writing output: {}", cli.out.display()))?;
57
58    Ok(())
59}
